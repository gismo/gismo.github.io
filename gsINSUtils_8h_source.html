26template<class T>
27void diagInvMatrix_into(const gsSparseMatrix<T>& mat, gsSparseMatrix<T>& diagInv, int repeat, bool lumping = false)
28{
29 GISMO_ENSURE(mat.nonZeros() != 0, "diagInvMatrix_into(): The matrix is empty!");
30
31 int varDofs = mat.rows();
32
33 diagInv.resize(repeat*varDofs, repeat*varDofs);
34 diagInv.reserve(gsVector<int>::Constant(diagInv.cols(), 1));
35
36 const gsSparseMatrix<T>* matPtr = &mat;
37 gsSparseMatrix<T> lumped(varDofs, varDofs);
38
39 if (lumping)
40 {
41 lumped.reserve(gsVector<int>::Constant(varDofs, 1));
42
43 for (int j = 0; j < varDofs; j++)
44 lumped.insert(j, j) = math::abs(mat.at(j, j)); // abs value because of "lumping" diag(A) in SIMPLE-type prec., does not change lumped mass matrix in IgA
45
46 for (int j = 0; j < varDofs; j++)
47 {
48 for (typename gsSparseMatrix<T>::InnerIterator it(mat, j); it; ++it)
49 {
50 int i = it.row();
51
52 if (i != j)
53 lumped.coeffRef(i, i) += math::abs(it.value());
54 }
55 }
56
57 matPtr = &lumped;
58 }
59
60 for (int i = 0; i < varDofs; i++)
61 {
62 T tmp = 1 / matPtr->coeff(i, i);
63
64 for (int s = 0; s < repeat; s++)
65 diagInv.coeffRef(i + s * varDofs, i + s * varDofs) = tmp;
66 }
67}

78template<class T>
79gsTensorBSpline<2, T> BSplineRectangle(int deg, const T llx, const T lly, const T a, const T b, int numSep = 0)
80{
81 gsKnotVector<T> kv(0, 1, numSep, deg + 1, deg); // first, last, num_inter, mult_end, mult_inter
82
83 int n = kv.size() - deg - 1;
84 gsMatrix<T> coef(n*n, 2);
85
86 for (int i = 0; i < n; i++)
87 for (int j = 0; j < n; j++)
88 coef.row(i + j*n) << llx + (i*a) / (deg*(numSep + 1)), lly + (j*b) / (deg*(numSep + 1));
89
90 return gsTensorBSpline<2, T>(kv, kv, coef);
91}

153template<class T> gsMultiPatch<T> BSplineStep2D(int deg, const T a, const T b, const T a_in, T h = 0, bool periodic = false)
154{
155 gsMultiPatch<T> mp;
156
157 if (h == 0)
158 h = b / 2;
159
160 mp.addPatch(BSplineRectangle(deg, 0.0, 0.0, a, h));
161 mp.addPatch(BSplineRectangle(deg, 0.0, h, a, b - h));
162 mp.addPatch(BSplineRectangle(deg, -a_in, h, a_in, b - h));
163
164 mp.addInterface(0, boundary::north, 1, boundary::south);
165 mp.addInterface(1, boundary::west, 2, boundary::east);
166
167 if(periodic)
168 mp.addInterface(0, boundary::south, 1, boundary::north);
169
170 mp.addAutoBoundaries();
171
172 return mp;
173}

236void defineBCs_cavity2D(gsBoundaryConditions<T>& bcInfo, const int np, std::vector<std::pair<int, boxSide> >& bndWall, std::string lidVel = "1")
237{
238 gsFunctionExpr<T> Uwall("0", "0", 2);
239 gsFunctionExpr<T> Ulid(lidVel, "0", 2);
240
241 for (int i = 1; i <= np; i++)
242 {
243 bcInfo.addCondition(np*np - i, boundary::north, condition_type::dirichlet, Ulid, 0);
244 bndWall.push_back(std::make_pair(np*np - i, boundary::north));
245 }
246
247 for (int i = 0; i < np; i++)
248 {
249 bcInfo.addCondition(i, boundary::south, condition_type::dirichlet, Uwall, 0);
250 bndWall.push_back(std::make_pair(i, boundary::south));
251 }
252
253 for (int i = 0; i < np; i++)
254 {
255 bcInfo.addCondition(i * np, boundary::west, condition_type::dirichlet, Uwall, 0);
256 bcInfo.addCondition((i + 1)*np - 1, boundary::east, condition_type::dirichlet, Uwall, 0);
257 bndWall.push_back(std::make_pair(i * np, boundary::west));
258 bndWall.push_back(std::make_pair((i + 1)*np - 1, boundary::east));
259 }
260}

417template< int d, class T> void refineLastKnotSpan(gsMultiBasis<T>& basis, int numRefine, int patch, int dir)
418{
419 const gsTensorBSplineBasis<d, T>* patchBasis = dynamic_cast<const gsTensorBSplineBasis<d, T>*>(&(basis.basis(patch)));
420 gsMatrix<T> box(d, 2);
421
422 for (int i = 0; i < numRefine; i++)
423 {
424 box.setZero();
425 int sizeKnots = patchBasis->knots(dir).size() - 1;
426 T lastKnot = patchBasis->knot(dir, sizeKnots);
427 T knot = patchBasis->knot(dir, sizeKnots - (patchBasis->degree(dir) + 1));
428 box(dir, 0) = knot;
429 box(dir, 1) = lastKnot;
430 basis.refine(patch, box);
431 }

441template<class T> void refineBasis_cavity2D(gsMultiBasis<T>& basis, int numRefine, int numRefineLocal, int numSep = 0)
442{
443 int npDir = math::sqrt(basis.nPieces());
444 int npRef = std::log2(npDir);
445 int sepRef = std::log2(numSep + 1);
446
447 for (int i = 0; i < numRefine - npRef - sepRef; ++i)

496template<class T> void refineBasis_step(gsMultiBasis<T>& basis, int numRefine, int numRefineWalls, int numRefineCorner, int numRefineU, real_t addRefPart, int dim, real_t a, real_t b, real_t c = 0.0)
497{
498 for (int i = 0; i < numRefine; ++i)
499 basis.uniformRefine();
500
501 gsMatrix<T> box(dim, 2);
502
503 int uRefine = math::floor(std::log2(a / b)) + 1 + numRefineU;
504 box.setZero();
505 box(0, 1) = 1;
506 for (int i = 0; i < uRefine; i++)
507 for (int p = 0; p < 2; p++)
508 basis.refine(p, box);
509
510 if (dim == 3)
511 {
512 int wRefine = math::floor(std::log2(c / b)) + 1;
513 box.setZero();
514 box(2, 1) = 1;
515 for (int i = 0; i < wRefine; i++)
516 for (int p = 0; p < 3; p++)
517 basis.refine(p, box);
518 }
519
520 box.setZero();
521 box(0,1) = addRefPart;
522 for (int p = 0; p < 2; p++)
523 basis.refine(p, box);
524
525 switch (dim)
526 {
527 case 2:
528 refineLocal_step<2, T>(basis, numRefineWalls, numRefineCorner);
529 break;
530 case 3:
531 refineLocal_step<3, T>(basis, numRefineWalls, numRefineCorner);
532 break;
533 default:
534 GISMO_ERROR("Wrong dimension!");
535 break;
536 }
537}
