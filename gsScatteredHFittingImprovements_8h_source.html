65void scatteredHFitting_improvements(gsTHBSplineBasis<2>& basis,
66 const gsMatrix<>& parameters,
67 const gsMatrix<>& points,
68 const int num_iterations,
69 const real_t threshold,
70 const real_t lambda,
71 const real_t alpha,
72 const int cp_magnitude,
73 const real_t hole,
74 const int n_loc,
75 const bool dirU,
76 const bool dirV,
77 const int circle,
78 const int strategy)
79{
80 gsMatrix<> coefficients;
81 gsMatrix<> old_coefficients;
82 std::vector<std::map<index_t, index_t> > tensorIndices; // tensorIndices[lvl] assigns the Hierarchical index to the tensor-product index of level lvl.
83 int dir = -1;
84
85 std::vector<index_t> enlargement;
86 std::vector<index_t> old_enlargement;
87
88 const bool is_periodic = periodic_surface(dirU, dirV);
89 if(is_periodic)
90 dir = get_periodic_dir(dirU, dirV);
91
92 for (int iter = 0; iter != num_iterations; iter++)
93 {
94 gsInfo << "\nIteration: " << iter + 1 << " / " << num_iterations << std::endl;
95
96 if(iter == 0)
97 {
98 gsStopwatch time;
99 time.restart();
100 if(is_periodic)
101 coefficients = get_coefficients_periodic(basis, parameters, points, iter, cp_magnitude, hole, lambda, dir, circle, enlargement);
102 else
103 coefficients = get_coefficients_not_scaled_stiff(basis, parameters, points, iter, cp_magnitude, hole, lambda, circle, enlargement);
104 time.stop();
105 gsInfo << "Fitting time: " << time << std::endl;
106 }
107 else
108 {
109 old_coefficients = coefficients;
110 old_enlargement = enlargement;
111 coefficients.resize(basis.size(), points.rows());
112 gsStopwatch time;
113 time.restart();
114 if(is_periodic)
115 coefficients = get_coefficients_periodic_opt(old_coefficients, tensorIndices, basis, parameters, points, iter, cp_magnitude, hole, lambda, dir, circle, old_enlargement, enlargement);
116 else
117 coefficients = get_coefficients_opt(old_coefficients, tensorIndices, basis, parameters, points, iter, cp_magnitude, hole, lambda, circle, old_enlargement, enlargement);
118 time.stop();
119 gsInfo << "Fitting time: " << time << std::endl;
120 }
121
122
123 gsTHBSpline<2> QI(basis, coefficients); // quasi interpolant
124
125 gsMatrix<> errors = compute_errors(QI, parameters, points);
126 real_t percentage = percentage_of_points_below_threshold(errors, threshold);
127
128 std::cout << "error percentage below threshold: " << 100 * percentage << "%\n";
129
130 // saving mesh
131 gsMesh<> mesh;
132 makeMesh<>(basis, mesh, 4);
133 std::string output = "iteration";
134 std::string prefix = output + internal::to_string(iter);
135
136 gsWriteParaview(mesh, prefix + "mesh");
137 QI.evaluateMesh(mesh);
138 gsWriteParaview(mesh, prefix + "geometry");
139 gsWriteParaview(QI, prefix + "thb", 10000, false, true);
140
141
142 // tensorBspline
143 gsTensorBSpline<2, real_t> tensorB;
144 QI.convertToBSpline(tensorB);
145 gsTensorBSplineBasis<2, real_t> tensBasis = tensorB.basis();
146 gsMesh<> TensorMesh;
147 makeMesh<>(tensBasis, TensorMesh, 4);
148 gsWriteParaview(TensorMesh, prefix + "tensor_mesh");
149 QI.evaluateMesh(TensorMesh);
150 gsWriteParaview(TensorMesh, prefix + "tensor_geometry");
151
152 std::cout << "Tensor DOFs: " << tensBasis.size() << std::endl;
153 std::cout << "THB DOFs: " << basis.size() << std::endl;
154
155 // Advanced visualisation with NX:
156 /*gsTHBSpline<2> copy(QI);
157 gsTensorBSpline<2> TP_spline;
158 copy.convertToBSpline(TP_spline);
159 extensions::gsWritePK_SHEET(TP_spline, "iteration" + std::to_string(iter));*/
160
161 if ((iter != num_iterations - 1) && (alpha < 1 - percentage))
162 {
163 // Fill tensorIndices before refinement.
164 tensorIndices.clear();
165 tensorIndices.resize(basis.getBases().size()+1); // +1, because there will be an extra level in the next step.
166 for(index_t k = 0; k != basis.size(); k++)
167 {
168 int lvl = basis.levelOf(k);
169 tensorIndices[lvl].insert(std::pair<index_t, index_t>(basis.flatTensorIndexOf(k), k));
170 }
171 if(is_periodic)
172 scatteredHFittingRefine_periodic(basis, parameters, errors, threshold, n_loc, dir, strategy, circle, enlargement);
173 else
174 scatteredHFittingRefine(basis, parameters, errors, threshold, n_loc, strategy, circle, enlargement);
175
176 if(old_coefficients.rows() == coefficients.rows())
177 {
178 std::cout << "No further refinement: alpha stopping criteria not satisfied. " << std::endl;
179 break;
180 }
181 }
182 else
183 {
184 break;
185 }
186 }
187}
