329 void calculate_voigt_stress(const gsMatrix<T>& tmp, short_t d, gsMatrix<T>& result) const
330 {
331 for (index_t k = 0; k < result.cols(); k++)
332 {
333 gsMatrix<T> S = tmp.reshapeCol(k, d, d); // in tensor notation
334 gsVector<T> S_voigt;
335 voigtStress(S_voigt, S); // Convert to Voigt notation
336 result.col(k) = S_voigt;
337 }
338 }
339
340 void calculate_voigt_strain(const gsMatrix<T>& tmp, short_t d, gsMatrix<T>& result) const
341 {
342 for (index_t k = 0; k < result.cols(); k++)
343 {
344 gsMatrix<T> E = tmp.reshapeCol(k, d, d); // in tensor notation
345 gsVector<T> E_voigt;
346 voigtStrain(E_voigt, E); // Convert to Voigt notation
347 result.col(k) = E_voigt;
348 }
349 }
