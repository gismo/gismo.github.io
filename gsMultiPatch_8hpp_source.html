358 /*
359  This is based on comparing a set of reference points of the patch
360  side and thus it implicitly assumes that the patch faces match
361 */
362 template<class T>
363 bool gsMultiPatch<T>::computeTopology( T tol, bool cornersOnly, bool)
364 {
365  BaseA::clearTopology();
366 
367  const size_t np = m_patches.size();
368  const index_t nCorP = 1 << m_dim; // corners per patch
369  const index_t nCorS = 1 << (m_dim-1); // corners per side
370 
371  gsMatrix<T> supp,
372  // Parametric coordinates of the reference points. These points
373  // are used to decide if two sides match.
374  // Currently these are the corner points and the side-centers
375  coor;
376  if (cornersOnly)
377  coor.resize(m_dim,nCorP);
378  else
379  coor.resize(m_dim,nCorP + 2*m_dim);
380 
381  gsVector<bool> boxPar(m_dim);
382 
383  // each matrix contains the physical coordinates of the reference points
384  std::vector<gsMatrix<T> > pCorners(np);
385 
386  std::vector<patchSide> pSide; // list of all candidate patchSides to compare
387  pSide.reserve(np * 2 * m_dim);
388 
389 //# pragma omp parallel for private(supp, boxPar, !coor)
390  for (size_t p=0; p<np; ++p)
391  {
392  supp = m_patches[p]->parameterRange(); // the parameter domain of patch i
393 
394  // Corners' parametric coordinates
395  for (boxCorner c=boxCorner::getFirst(m_dim); c<boxCorner::getEnd(m_dim); ++c)
396  {
397  boxPar = c.parameters(m_dim);
398  for (index_t i=0; i<m_dim;++i)
399  coor(i,c-1) = boxPar(i) ? supp(i,1) : supp(i,0);
400  }
401 
402  if (!cornersOnly)
403  {
404  // Sides' centers parametric coordinates
405  index_t l = nCorP;
406  for (boxSide c=boxSide::getFirst(m_dim); c<boxSide::getEnd(m_dim); ++c)
407  {
408  const index_t dir = c.direction();
409  const index_t s = static_cast<index_t>(c.parameter());// 0 or 1
410 
411  for (index_t i=0; i<m_dim;++i)
412  coor(i,l) = ( dir==i ? supp(i,s) :
413  (supp(i,1)+supp(i,0))/2.0 );
414  l++;
415  }
416  }
417 
418  // Evaluate the patch on the reference points
419  m_patches[p]->eval_into(coor,pCorners[p]);
420 
421  // Add all the patchSides of this patch to the candidate list
422  for (boxSide bs=boxSide::getFirst(m_dim); bs<boxSide::getEnd(m_dim); ++bs)
423  pSide.push_back(patchSide(p,bs));
424  }
425 
426  gsVector<index_t> dirMap(m_dim);
427  gsVector<bool> matched(nCorS), dirOr(m_dim);
428  std::vector<boxCorner> cId1, cId2;
429  cId1.reserve(nCorS);
430  cId2.reserve(nCorS);
431 
432  std::set<index_t> found;
433  for (size_t sideind=0; sideind<pSide.size(); ++sideind)
434  {
435  const patchSide & side = pSide[sideind];
436  for (size_t other=sideind+1; other<pSide.size(); ++other)
437  {
438  side .getContainedCorners(m_dim,cId1);
439  pSide[other].getContainedCorners(m_dim,cId2);
440  matched.setConstant(false);
441 
442  // Check whether the side center matches
443  if (!cornersOnly)
444  if ( ( pCorners[side.patch ].col(nCorP+side-1 ) -
445  pCorners[pSide[other].patch].col(nCorP+pSide[other]-1)
446  ).norm() >= tol )
447  continue;
448 
449  //t-junction
450  // check for matching vertices else
451  // invert the vertices of first side on the second and vise-versa
452  // if at least one vertex is found (at most 2^(d-1)), mark as interface
453 
454  // Check whether the vertices match and compute direction
455  // map and orientation
456  if ( matchVerticesOnSide( pCorners[side.patch] , cId1, 0,
457  pCorners[pSide[other].patch], cId2,
458  matched, dirMap, dirOr, tol ) )
459  {
460  dirMap(side.direction()) = pSide[other].direction();
461  dirOr (side.direction()) = !( side.parameter() == pSide[other].parameter() );
462  BaseA::addInterface( boundaryInterface(side, pSide[other], dirMap, dirOr));
463  found.insert(sideind);
464  found.insert(other);
465  }
466  }
467  }
468 
469  index_t k = 0;
470  found.insert(found.end(), pSide.size());
471  for (const auto & s : found)
472  {
473  for (;k<s;++k)
474  BaseA::addBoundary( pSide[k] );
475  ++k;
476  }
477 
478  return true;
479 }

493 template <class T>
494 bool gsMultiPatch<T>::matchVerticesOnSide (
495  const gsMatrix<T> &cc1, const std::vector<boxCorner> &ci1, index_t start,
496  const gsMatrix<T> &cc2, const std::vector<boxCorner> &ci2,
497  const gsVector<bool> &matched, gsVector<index_t> &dirMap,
498  gsVector<bool> &dirO, T tol, index_t reference)
499 {
500  const bool computeOrientation = !(start&(start-1)) && (start != 0); // true if start is a power of 2
501  const bool setReference = start==0; // if we search for the first point then we set the reference
502 
503  const short_t dim = static_cast<short_t>(cc1.rows());
504 
505  index_t o_dir = 0, d_dir = 0;
506 
507  gsVector<bool> refPar, newPar, newMatched;
508 
509  if (computeOrientation)
510  {
511  // o_dir is the only direction in which the parameters differ between start and 0
512  const gsVector<bool> parStart = ci1[start].parameters(dim);
513  const gsVector<bool> parRef = ci1[0].parameters(dim);
514  for (; o_dir<dim && parStart(o_dir)==parRef(o_dir) ;) ++o_dir;
515  }
516 
517  if (!setReference)
518  refPar = ci2[reference].parameters(dim);
519 
520  for (size_t j=0;j<ci2.size();++j)
521  {
522  if( !matched(j) && (cc1.col(ci1[start]-1)-cc2.col(ci2[j]-1)).norm() < tol )
523  {
524  index_t newRef = (setReference) ? j : reference;
525  if (computeOrientation)
526  {
527  index_t count=0;
528  d_dir = 0;
529  newPar = ci2[j].parameters(dim);
530  for (index_t i=0; i< newPar.rows();++i)
531  {
532  if ( newPar(i)!=refPar(i) )
533  {
534  d_dir=i;
535  ++count;
536  }
537  }
538  if (count != 1)
539  {
540  // the match is wrong: we are mapping an edge to a diagonal
541  continue;
542  }
543  dirMap(o_dir) = d_dir;
544  dirO (o_dir) = (static_cast<index_t>(j) > reference);
545  }
546  if ( start + 1 == static_cast<index_t>( ci1.size() ) )
547  {
548  // we matched the last vertex, we are done
549  return true;
550  }
551  newMatched = matched;
552  newMatched(j) = true;
553  if (matchVerticesOnSide( cc1,ci1,start+1,cc2,ci2,newMatched,dirMap,dirO, tol,newRef))
554  return true;
555  }
556  }
557 
558  return false;
559 }

562 template<class T>
563 void gsMultiPatch<T>::closeGaps(T tol)
564 {
565  const T tol2 = tol*tol;
566  gsMatrix<index_t> bdr1, bdr2; // indices of the boundary control points
567 
568  // Create a map which assigns to all meeting patch-local indices a
569  // unique global id
570  const index_t sz = m_patches.size();
571  gsVector<index_t> patchSizes(sz);
572  for (index_t i=0; i!= sz; ++i)
573  patchSizes[i] = m_patches[i]->coefsSize();
574 
575  gsDofMapper mapper(patchSizes);
576 
577  for ( iiterator it = iBegin(); it != iEnd(); ++it ) // for all interfaces
578  {
579  const gsGeometry<T> & p1 = *m_patches[it->first() .patch];
580  const gsGeometry<T> & p2 = *m_patches[it->second().patch];
581 
582  // Grab boundary control points in matching configuration
583  p1.basis().matchWith(*it, p2.basis(), bdr1, bdr2);
584 
585  bool warn = true;
586  //mapper.matchDofs(it->first().patch, bdr1, it->second().patch, bdr2);
587  for (index_t i = 0; i!= bdr1.size(); ++i )
588  {
589  if ( ( p1.coef(bdr1(i)) - p2.coef(bdr2(i)) ).squaredNorm() > tol2 )
590  {
591  if (warn)
592  {
593  gsWarn<<"Big gap detected between patches "<< it->first().patch
594  <<" and "<<it->second().patch <<"\n";
595  warn = false;
596  }
597  }
598  else
599  // Match the dofs on the interface
600  mapper.matchDof(it->first().patch, bdr1(i,0), it->second().patch, bdr2(i,0) );
601  }
602  }
603 
604  // Finalize the mapper. At this point all patch-local dofs are
605  // mapped to unique global indices
606  mapper.finalize();
607 
608  gsMatrix<T> meanVal;
609  std::vector<std::pair<index_t,index_t> > dof;
610  const index_t start = mapper.freeSize() - mapper.coupledSize();
611  const index_t end = mapper.freeSize();
612 
613  for (index_t i = start; i!= end; ++i) // For all coupled DoFs
614  {
615  // Get the preimages of this global dof (as pairs (patch,index) )
616  mapper.preImage(i, dof);
617 
618  // Compute the mean value
619  meanVal = m_patches[dof.front().first]->coef(dof.front().second);
620  for (size_t k = 1; k!=dof.size(); ++k)
621  meanVal += m_patches[dof[k].first]->coef(dof[k].second);
622  meanVal.array() /= dof.size();
623 
624  // Set involved control points equal to their average value
625  for (size_t k = 0; k!=dof.size(); ++k)
626  m_patches[dof[k].first]->coef(dof[k].second) = meanVal;
627  }
628 }

672 {
673  gsMultiBasis<T> multiBasis(*this);
674  bool changed = false;
675 
676  std::vector<index_t> refEltsFirst;
677  std::vector<index_t> refEltsSecond;

687  break;
688  default:
689  GISMO_ASSERT(false,"wrong dimension");

691 
692  // ...and if there are any found, refine the bases accordingly

694  {
695  if( refEltsFirst.size() > 0 )
696  {

698  patch(pi).basis().refineElements_withCoefs( patch(pi).coefs(), refEltsFirst );

701  {

703  patch(pi).basis().refineElements_withCoefs( patch(pi).coefs(), refEltsSecond );
704  }

795  std::pair<index_t,gsVector<T> > & result,
796  const T accuracy) const
797 {
798  GISMO_ASSERT( pt.rows() == targetDim(), "Invalid input point." <<
799  pt.rows() <<"!="<< targetDim() );
800 
801  gsVector<T> tmp;
802 
803 #ifndef _MSC_VER
804 # pragma omp declare reduction(minimum : struct __closestPointHelper : omp_out = (omp_in.dist < omp_out.dist ? omp_in : omp_out) )
805  struct __closestPointHelper cph;
806 # pragma omp parallel for default(shared) private(tmp) reduction(minimum:cph) //OpenMP 4.0, will not work on VS2019
807 #else
808  struct __closestPointHelper cph;
809 #endif
810  for (size_t k = 0; k < m_patches.size(); ++k)
811  {
812  // possible improvement: approximate dist: eval patch on a

814

819  cph.pid = k;
820  cph.preim = tmp;
821  }
822  }
