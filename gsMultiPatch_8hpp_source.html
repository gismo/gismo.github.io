367 /*
368  This is based on comparing a set of reference points of the patch
369  side and thus it implicitly assumes that the patch faces match
370 */
371 template<class T>
372 bool gsMultiPatch<T>::computeTopology( T tol, bool cornersOnly, bool)
373 {
374  BaseA::clearTopology();
375 
376  const size_t np = m_patches.size();
377  const index_t nCorP = 1 << m_dim; // corners per patch
378  const index_t nCorS = 1 << (m_dim-1); // corners per side
379 
380  gsMatrix<T> supp,
381  // Parametric coordinates of the reference points. These points
382  // are used to decide if two sides match.
383  // Currently these are the corner points and the side-centers
384  coor;
385  if (cornersOnly)
386  coor.resize(m_dim,nCorP);
387  else
388  coor.resize(m_dim,nCorP + 2*m_dim);
389 
390  gsVector<bool> boxPar(m_dim);
391 
392  // each matrix contains the physical coordinates of the reference points
393  std::vector<gsMatrix<T> > pCorners(np);
394 
395  std::vector<patchSide> pSide; // list of all candidate patchSides to compare
396  pSide.reserve(np * 2 * m_dim);
397 
398 //# pragma omp parallel for private(supp, boxPar, !coor)
399  for (size_t p=0; p<np; ++p)
400  {
401  supp = m_patches[p]->parameterRange(); // the parameter domain of patch i
402 
403  // Corners' parametric coordinates
404  for (boxCorner c=boxCorner::getFirst(m_dim); c<boxCorner::getEnd(m_dim); ++c)
405  {
406  boxPar = c.parameters(m_dim);
407  for (index_t i=0; i<m_dim;++i)
408  coor(i,c-1) = boxPar(i) ? supp(i,1) : supp(i,0);
409  }
410 
411  if (!cornersOnly)
412  {
413  // Sides' centers parametric coordinates
414  index_t l = nCorP;
415  for (boxSide c=boxSide::getFirst(m_dim); c<boxSide::getEnd(m_dim); ++c)
416  {
417  const index_t dir = c.direction();
418  const index_t s = static_cast<index_t>(c.parameter());// 0 or 1
419 
420  for (index_t i=0; i<m_dim;++i)
421  coor(i,l) = ( dir==i ? supp(i,s) :
422  (supp(i,1)+supp(i,0))/2.0 );
423  l++;
424  }
425  }
426 
427  // Evaluate the patch on the reference points
428  m_patches[p]->eval_into(coor,pCorners[p]);
429 
430  // Add all the patchSides of this patch to the candidate list
431  for (boxSide bs=boxSide::getFirst(m_dim); bs<boxSide::getEnd(m_dim); ++bs)
432  pSide.push_back(patchSide(p,bs));
433  }
434 
435  gsVector<index_t> dirMap(m_dim);
436  gsVector<bool> matched(nCorS), dirOr(m_dim);
437  std::vector<boxCorner> cId1, cId2;
438  cId1.reserve(nCorS);
439  cId2.reserve(nCorS);
440 
441  std::set<index_t> found;
442  for (size_t sideind=0; sideind<pSide.size(); ++sideind)
443  {
444  const patchSide & side = pSide[sideind];
445  for (size_t other=sideind+1; other<pSide.size(); ++other)
446  {
447  side .getContainedCorners(m_dim,cId1);
448  pSide[other].getContainedCorners(m_dim,cId2);
449  matched.setConstant(false);
450 
451  // Check whether the side center matches
452  if (!cornersOnly)
453  if ( ( pCorners[side.patch ].col(nCorP+side-1 ) -
454  pCorners[pSide[other].patch].col(nCorP+pSide[other]-1)
455  ).norm() >= tol )
456  continue;
457 
458  //t-junction
459  // check for matching vertices else
460  // invert the vertices of first side on the second and vise-versa
461  // if at least one vertex is found (at most 2^(d-1)), mark as interface
462 
463  // Check whether the vertices match and compute direction
464  // map and orientation
465  if ( matchVerticesOnSide( pCorners[side.patch] , cId1, 0,
466  pCorners[pSide[other].patch], cId2,
467  matched, dirMap, dirOr, tol ) )
468  {
469  dirMap(side.direction()) = pSide[other].direction();
470  dirOr (side.direction()) = !( side.parameter() == pSide[other].parameter() );
471  BaseA::addInterface( boundaryInterface(side, pSide[other], dirMap, dirOr));
472  found.insert(sideind);
473  found.insert(other);
474  }
475  }
476  }
477 
478  index_t k = 0;
479  found.insert(found.end(), pSide.size());
480  for (const auto & s : found)
481  {
482  for (;k<s;++k)
483  BaseA::addBoundary( pSide[k] );
484  ++k;
485  }
486 
487  return true;
488 }

502 template <class T>
503 bool gsMultiPatch<T>::matchVerticesOnSide (
504  const gsMatrix<T> &cc1, const std::vector<boxCorner> &ci1, index_t start,
505  const gsMatrix<T> &cc2, const std::vector<boxCorner> &ci2,
506  const gsVector<bool> &matched, gsVector<index_t> &dirMap,
507  gsVector<bool> &dirO, T tol, index_t reference)
508 {
509  const bool computeOrientation = !(start&(start-1)) && (start != 0); // true if start is a power of 2
510  const bool setReference = start==0; // if we search for the first point then we set the reference
511 
512  const short_t dim = static_cast<short_t>(cc1.rows());
513 
514  index_t o_dir = 0, d_dir = 0;
515 
516  gsVector<bool> refPar, newPar, newMatched;
517 
518  if (computeOrientation)
519  {
520  // o_dir is the only direction in which the parameters differ between start and 0
521  const gsVector<bool> parStart = ci1[start].parameters(dim);
522  const gsVector<bool> parRef = ci1[0].parameters(dim);
523  for (; o_dir<dim && parStart(o_dir)==parRef(o_dir) ;) ++o_dir;
524  }
525 
526  if (!setReference)
527  refPar = ci2[reference].parameters(dim);
528 
529  for (size_t j=0;j<ci2.size();++j)
530  {
531  if( !matched(j) && (cc1.col(ci1[start]-1)-cc2.col(ci2[j]-1)).norm() < tol )
532  {
533  index_t newRef = (setReference) ? j : reference;
534  if (computeOrientation)
535  {
536  index_t count=0;
537  d_dir = 0;
538  newPar = ci2[j].parameters(dim);
539  for (index_t i=0; i< newPar.rows();++i)
540  {
541  if ( newPar(i)!=refPar(i) )
542  {
543  d_dir=i;
544  ++count;
545  }
546  }
547  if (count != 1)
548  {
549  // the match is wrong: we are mapping an edge to a diagonal
550  continue;
551  }
552  dirMap(o_dir) = d_dir;
553  dirO (o_dir) = (static_cast<index_t>(j) > reference);
554  }
555  if ( start + 1 == static_cast<index_t>( ci1.size() ) )
556  {
557  // we matched the last vertex, we are done
558  return true;
559  }
560  newMatched = matched;
561  newMatched(j) = true;
562  if (matchVerticesOnSide( cc1,ci1,start+1,cc2,ci2,newMatched,dirMap,dirO, tol,newRef))
563  return true;
564  }
565  }
566 
567  return false;
568 }

571 template<class T>
572 void gsMultiPatch<T>::closeGaps(T tol)
573 {
574  gsDofMapper mapper = getMapper(tol);
575 
576  gsMatrix<T> meanVal;
577  std::vector<std::pair<index_t,index_t> > dof;
578  const index_t start = mapper.freeSize() - mapper.coupledSize();
579  const index_t end = mapper.freeSize();
580 
581  for (index_t i = start; i!= end; ++i) // For all coupled DoFs
582  {
583  // Get the preimages of this global dof (as pairs (patch,index) )
584  mapper.preImage(i, dof);
585 
586  // Compute the mean value
587  meanVal = m_patches[dof.front().first]->coef(dof.front().second);
588  for (size_t k = 1; k!=dof.size(); ++k)
589  meanVal += m_patches[dof[k].first]->coef(dof[k].second);
590  meanVal.array() /= dof.size();
591 
592  // Set involved control points equal to their average value
593  for (size_t k = 0; k!=dof.size(); ++k)
594  m_patches[dof[k].first]->coef(dof[k].second) = meanVal;
595  }
596 }

598 template<class T>
599 gsDofMapper gsMultiPatch<T>::getMapper(T tol) const
600 {
601  const T tol2 = tol*tol;
602  gsMatrix<index_t> bdr1, bdr2; // indices of the boundary control points
603 
604  // Create a map which assigns to all meeting patch-local indices a
605  // unique global id
606  const index_t sz = m_patches.size();
607  gsVector<index_t> patchSizes(sz);
608  for (index_t i=0; i!= sz; ++i)
609  patchSizes[i] = m_patches[i]->coefsSize();
610 
611  gsDofMapper mapper(patchSizes);
612 
613  for ( const_iiterator it = iBegin(); it != iEnd(); ++it ) // for all interfaces
614  {
615  const gsGeometry<T> & p1 = *m_patches[it->first() .patch];
616  const gsGeometry<T> & p2 = *m_patches[it->second().patch];
617 
618  // Grab boundary control points in matching configuration
619  p1.basis().matchWith(*it, p2.basis(), bdr1, bdr2);
620 
621  bool warn = true;
622  //mapper.matchDofs(it->first().patch, bdr1, it->second().patch, bdr2);
623  for (index_t i = 0; i!= bdr1.size(); ++i )
624  {
625  if ( ( p1.coef(bdr1(i)) - p2.coef(bdr2(i)) ).squaredNorm() > tol2 )
626  {
627  if (warn)
628  {
629  gsWarn<<"Big gap detected between patches "<< it->first().patch
630  <<" and "<<it->second().patch <<"\n";
631  warn = false;
632  }
633  }
634  else
635  // Match the dofs on the interface
636  mapper.matchDof(it->first().patch, bdr1(i,0), it->second().patch, bdr2(i,0) );
637  }
638  }
639 
640  // Finalize the mapper. At this point all patch-local dofs are
641  // mapped to unique global indices
642  mapper.finalize();
643  return mapper;
644 }

1003 template<class T>
1004 std::map< std::array<size_t, 4>, internal::ElementBlock> gsMultiPatch<T>::BezierOperator() const
1005 {
1006  GISMO_ENSURE( 2==domainDim(), "Anything other than bivariate splines is not yet supported!");
1007 
1008  // Loop over all the elements of the given multipatch and collect all relevant
1009  // information in ElementBlocks. These will be grouped in a std::map
1010  // with respect to the number of active basis functions ( = NN/NCV )
1011  // of each Bezier element
1012  std::map<std::array<size_t, 4>, internal::ElementBlock> ElementBlocks;
1013 
1014  index_t NN; // Number of control points of the Bezier element
1015  gsMatrix<index_t> localActives, globalActives; // Active basis functions
1016  gsDofMapper mapper = getMapper((T)1e-7);
1017 
1018  gsMatrix<T> quPoints, values;
1019  gsVector<T> quWeights;
1020 
1021  for (size_t p=0; p<nPatches(); ++p)
1022  {
1023  gsBasis<T> * basis = & patch(p).basis();
1024 
1025  // Create the Bezier Basis
1026  gsKnotVector<T> kv1;
1027  kv1.initClamped(basis->degree(0));
1028  gsKnotVector<T> kv2;
1029  kv2.initClamped(basis->degree(1));
1030  gsTensorBSplineBasis<2,T> bezBasis(kv1,kv2);
1031  gsMatrix<> res;
1032 
1033  // Initialize the quadrature rule that will be used for fitting
1034  // the given basis with the Bezier basis
1035  gsVector<index_t, 2> numNodes;
1036  numNodes << basis->degree(0)+1, basis->degree(1)+1 ;
1037  typename gsNewtonCotesRule<T>::uPtr QuRule;
1038  QuRule = gsNewtonCotesRule<T>::make(numNodes);
1039 
1040  // Initialize an iterator over all the elements of the given basi
1041  typename gsBasis<T>::domainIter domIt = basis->makeDomainIterator();
1042 
1043 
1044  // Calculate the collocation matrix of the Bezier Basis
1045  // It will be used to fit the Bez. Basis to the original basis' elements.
1046  gsMatrix<T> Bd = bezBasis.collocationMatrix(bezBasis.anchors());
1047  auto solver = Bd.fullPivLu();
1048 
1049  std::array<size_t, 4> key;
1050  for (; domIt->good(); domIt->next() )
1051  {
1052  localActives = basis->active( domIt->center );
1053  globalActives.resizeLike(localActives);
1054  // Map every local active basis function to the global numbering
1055  for (index_t i=0; i<localActives.rows(); ++i)
1056  globalActives.at(i) = mapper.index(localActives.at(i), p);
1057 
1058 
1059  key[0] = globalActives.rows();
1060  key[1] = basis->degree(0);
1061  key[2] = basis->degree(1);
1062  key[3] = 0; // TODO: if implemented for trivariates fix this
1063  NN = localActives.size();
1064  ElementBlocks[key].numElements += 1; // Increment the Number of Elements contained in the ElementBlock
1065  ElementBlocks[key].actives.push_back(globalActives); // Append the active basis functions ( = the Node IDs ) for this element.
1066  ElementBlocks[key].PR = basis->degree(0);
1067  ElementBlocks[key].PS = basis->degree(1);
1068  ElementBlocks[key].PT = 0; // TODO: if implemented for trivariates fix this
1069 
1070  // Map the quadrature points to the current element.
1071  QuRule->mapTo( domIt->lowerCorner(), domIt->upperCorner(), quPoints, quWeights);
1072  basis->source().eval_into(quPoints, values); // Evaluate given basis at the mapped quadrature points
1073  // Append the local Bezier Extraction matrix to the ElementBlock.coefVectors
1074  ElementBlocks[key].coefVectors.push_back(solver.solve(values.transpose()).transpose());
1075  }
1076  }
1077 
1078  return ElementBlocks;
1079 }

1082 template<class T>
1083 gsMultiPatch<T> gsMultiPatch<T>::extractBezier() const
1084 {
1085  GISMO_ENSURE( 2==domainDim(), "Anything other than bivariate splines is not yet supported!");
1086  std::map<std::array<size_t, 4>, internal::ElementBlock> ElementBlocks = BezierOperator();
1087  gsMultiPatch<T> result;
1088 
1089  // Get the map from patch-local to global indexing for the multi-patch
1090  gsDofMapper mapper = this->getMapper((T)1e-4);
1091 
1092  // Get global coefficients of the multi patch. (i.e. without duplicates on the interfaces)
1093  gsMatrix<T> globalCoefs(mapper.size(), this->coefs().cols());
1094  globalCoefs.setZero();
1095  gsMatrix<T> globalWeights(mapper.size(), 1);
1096  globalWeights.setOnes();
1097 
1098  // Loop over all patches
1099  for (index_t p = 0; p != this->nPatches(); p++)
1100  {
1101  for (index_t i=0; i != this->patch(p).coefs().rows(); ++i) // For every control point
1102  {
1103  globalCoefs.row(mapper.index(i,p)).leftCols(this->geoDim()) = this->patch(p).coefs().row(i);
1104  if (this->basis(p).isRational())
1105  globalWeights(mapper.index(i,p)) = this->basis(p).weights().at(i);
1106  }
1107  }
1108 
1109  gsMatrix<> newCoefs, newWeights;
1110  for (auto const& pair : ElementBlocks)
1111  {
1112  internal::ElementBlock ElBlock = pair.second;
1113 
1114  gsKnotVector<> kv1(0,1,0,ElBlock.PR+1);
1115  gsKnotVector<> kv2(0,1,0,ElBlock.PS+1);
1116  // coefs.setZero( (ElBlock.PR+1)*(ElBlock.PS+1), controlPoints.cols()-1 );
1117 
1118  // Loop over all elements of the Element Block
1119  auto Ait = ElBlock.actives.begin(); // Actives Iterator
1120  auto Cit = ElBlock.coefVectors.begin(); // Coefficients Iteratos
1121 
1122  for(; Ait != ElBlock.actives.end() && Cit != ElBlock.coefVectors.end(); ++Ait, ++Cit)
1123  {
1124  // If the weights are not all equal (Rational)
1125  if ( (globalWeights(Ait->asVector(),0).array() != globalWeights(Ait->asVector()(0),0)).any())
1126  {
1127  // As per Borden et al. 2010 "Isogeometric finite element data structures
1128  // based on Bézier extraction of NURBS", eq (79);
1129  newWeights = Cit->transpose() * globalWeights(Ait->asVector(),0);
1130  newCoefs = Cit->transpose() * globalWeights(Ait->asVector(),0).asDiagonal() * globalCoefs(Ait->asVector(),gsEigen::all);
1131  newCoefs = newCoefs.array().colwise() / newWeights.col(0).array();
1132 
1133  gsTensorNurbs<2> bezier(kv1,kv1, newCoefs, newWeights);
1134  result.addPatch(bezier);
1135  }
1136  else // If all weights are equal (Polynomial)
1137  {
1138  gsTensorBSpline<2> bezier(kv1,kv2, Cit->transpose() * globalCoefs(Ait->asVector(),gsEigen::all));
1139  result.addPatch(bezier);
1140  }
1141  // bezier extraction operator * original control points
1142  }
1143  }
1144 
1145  // result.computeTopology();
1146  return result;
1147 }
