477  void embed(index_t N, bool pad_right = true)
478  {
479  GISMO_ASSERT( N > 0, "Embed dimension must be positive");
480 
481  const index_t nc = N - m_coefs.cols();
482  if ( nc == 0 ) return;
483 
484  if (!pad_right && nc<0)
485  m_coefs.leftCols(N) = m_coefs.rightCols(N);
486  m_coefs.conservativeResize(gsEigen::NoChange, N);
487 
488  if ( nc > 0 )
489  {
490  if (pad_right)
491  m_coefs.rightCols(nc).setZero();
492  else
493  {
494  m_coefs.rightCols(N-nc) = m_coefs.leftCols(N-nc);
495  m_coefs.leftCols(nc).setZero();
496  }
497  }
498 # ifndef NDEBUG
499  else gsWarn<<"Coefficients projected (deleted)..\n";
500 # endif
501  }
