20// Indices of the Voigt notation
21inline short_t voigt(short_t dim, short_t I, short_t J)
22{
23 if (dim == 2)
24 switch(I)
25 {
26 case 0: return J == 0 ? 0 : 0;
27 case 1: return J == 0 ? 1 : 1;
28 case 2: return J == 0 ? 0 : 1;
29 }
30 else if (dim == 3)
31 switch (I)
32 {
33 case 0: return J == 0 ? 0 : 0;
34 case 1: return J == 0 ? 1 : 1;
35 case 2: return J == 0 ? 2 : 2;
36 case 3: return J == 0 ? 0 : 1;
37 case 4: return J == 0 ? 1 : 2;
38 case 5: return J == 0 ? 0 : 2;
39 }
40 GISMO_ERROR("voigt notation indices error");
41}

43// construct a fourth order symmetric identity tensor C based on a second order symmetric tensor R
44// C_ijkl = (R_ik*R_jl + R_il*R_jk)/2 in Voigt notation
45template <class T>
46inline void symmetricIdentityTensor(gsMatrix<T> & C, const gsMatrix<T> & R)
47{
48 short_t dim = R.cols();
49 short_t dimTensor = dim*(dim+1)/2;
50 C.setZero(dimTensor,dimTensor);
51 // componentwise definition
52 for (short_t i = 0; i < dimTensor; ++i)
53 for (short_t j = 0; j < dimTensor; ++j)
54 C(i,j) = (R(voigt(dim,i,0),voigt(dim,j,0))*R(voigt(dim,i,1),voigt(dim,j,1)) +
55 R(voigt(dim,i,0),voigt(dim,j,1))*R(voigt(dim,i,1),voigt(dim,j,0)));
56}

59template <class T> //template it with dimension
60inline void deviatoricTensor(gsMatrix<T> & C, const gsMatrix<T> & R)
61{
62 short_t dim = R.cols();
63 short_t dimTensor = dim*(dim+1)/2;
64 C.setZero(dimTensor,dimTensor);
65
66 C.block(0,0,dim,dim).setConstant(-1.0/3.0);
67 C.block(0,0,dim,dim).diagonal().setConstant(2.0/3.0);
68 C.block(dim,dim,dimTensor-dim,dimTensor-dim).diagonal().setConstant(0.5);
69
70 // if (dim == 2)
71 // {
72 // C <<2.0/3.0, -1.0/3.0, 0.,
73 // -1.0/3.0, 2.0/3.0, 0.,

88// construct a fourth order matrix-trace tensor C based on two second order symmetric tensors R and S
89// C_ijkl = R_ij*S_kl in Voigt notation
90template <class T>
91inline void matrixTraceTensor(gsMatrix<T> & C, const gsMatrix<T> & R, const gsMatrix<T> & S)
92{
93 short_t dim = R.cols();
94 short_t dimTensor = dim*(dim+1)/2;
95 C.setZero(dimTensor,dimTensor);
96 // componentwise definition
97 for (short_t i = 0; i < dimTensor; ++i)
98 for (short_t j = 0; j < dimTensor; ++j)
99 C(i,j) = R(voigt(dim,i,0),voigt(dim,i,1))*S(voigt(dim,j,0),voigt(dim,j,1));
100}

114template <class T>
115inline void tensorStress(const index_t dim, const gsMatrix<T> & Svec, gsMatrix<T> & S)
116{
117 if (dim == 2)
118 {
119 S.setZero(dim,dim);
120 S(0,0) = Svec(0,0);
121 S(1,1) = Svec(1,0);
122 S(0,1) = Svec(2,0);
123 S(1,0) = Svec(2,0);
124 }
125 else if (dim == 3)
126 {
127 S.setZero(dim,dim);
128 S(0,0) = Svec(0,0);
129 S(1,1) = Svec(1,0);
130 S(2,2) = Svec(2,0);
131
132 S(0,1) = Svec(3,0); // tao_xy
133 S(1,0) = Svec(3,0); // tao_xy
134
135 S(0,2) = Svec(4,0); // tao_yz
136 S(2,0) = Svec(4,0); // tao_yz
137

143// transform strain tensor E to a vector in Voigt notation
144template <class T>
145inline void voigtStrain(gsVector<T> & Evec, const gsMatrix<T> & E)
146{
147 short_t dim = E.cols();
148 //gsDebugVar(dim);
149 short_t dimTensor = dim*(dim+1)/2;
150 Evec.resize(dimTensor);
151 for (short i = 0; i < dimTensor; ++i)
152 if (voigt(dim,i,0) != voigt(dim,i,1))
153 Evec(i) = E(voigt(dim,i,0),voigt(dim,i,1)) + E(voigt(dim,i,1),voigt(dim,i,0)); // off-diagonal terms
154 else
155 Evec(i) = E(voigt(dim,i,0),voigt(dim,i,1)); // diagonal terms
156}

158// auxiliary matrix B such that E:S = B*Svec in the weak form
159// (see Bernal, Calo, Collier, et. at., "PetIGA", ICCS 2013, p. 1610)
160template <class T>
161inline void setB( gsMatrix<T> & B, const gsMatrix<T> & F, const gsVector<T> & bGrad)
162{
163 short_t dim = F.cols();
164 short_t dimTensor = dim*(dim+1)/2;
165 B.resize(dimTensor,dim);
166
167 for (short_t j = 0; j < dim; ++j)
168 {
169 for (short_t i = 0; i < dim; ++i)
170 B(i,j) = F(j,i) * bGrad(i);
171 if (dim == 2)
172 B(2,j) = F(j,0) * bGrad(1) + F(j,1) * bGrad(0);
173 if (dim == 3)
174 for (short_t i = 0; i < dim; ++i)
175 {
176 short_t k = (i+1)%dim;
177 B(i+dim,j) = F(j,i) * bGrad(k) + F(j,k) * bGrad(i);
178 }
179 }
