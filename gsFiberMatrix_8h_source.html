213 void conservativeResize(index_t newRows, index_t newCols)
214 {
215 if (!IsRowMajor) std::swap(newRows,newCols);
216
217 const index_t oldRows = fibers();
218
219 // delete any fibers which will be removed, if any
220 for (index_t i = newRows; i < oldRows; ++i)
221 delete m_fibers[i];
222
223 m_fibers.resize(newRows);
224
225 // allocate newly added fibers, if any
226 for (index_t i = oldRows; i < newRows; ++i)
227 m_fibers[i] = new Fiber(newCols);
228
229 const index_t m = std::min(oldRows, newRows);
230 for (index_t i = 0; i < m; ++i)
231 m_fibers[i]->conservativeResize(newCols);
232 }

270 template <class Derived>
271 void toSparseMatrix(gsEigen::SparseMatrixBase<Derived>& m) const
272 {
273 m.derived().resize( rows(), cols() );
274 m.derived().reserve( nonZeros() );
275 for (index_t i = 0; i < fibers(); ++i)
276 {
277 for (typename Fiber::InnerIterator it(*m_fibers[i]); it; ++it)
278 m.derived().insert(IsRowMajor?i:it.index(), IsRowMajor?it.index():i) = it.value();
279 }
280 m.derived().makeCompressed();
281 }
