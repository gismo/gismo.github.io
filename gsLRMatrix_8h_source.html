101 // row/col indices and first component value
102 for(index_t i = l-1; i>=0; --i)
103 {
104 nr = nr * m_cwise[i].rows() + m_rit[i].index();
105 nc = nc * (m_col.upper()[i]+1) + m_col->at(i);
106 tmp *= m_rit[i].value();
107 }
108
109 // add up upto rank
110 for( size_t k = n; k!=m_rit.size(); k+=n )
111 {
112 T tmpr = m_rit[k].value();
113 for(index_t i = 1; i!=n; ++i)
114 tmpr *= m_rit[k+i].value();
115 tmp += tmpr;
116 }
117
118 return tmp;
119 }
