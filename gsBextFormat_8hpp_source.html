34template<class T>
35void exportBext(gsMultiPatch<T> & mPatch, std::string filename)
36{
37 GISMO_ENSURE(!filename.empty(), "No output filename provided.");
38 // Get the map from patch-local to global indexing for the multi-patch
39 gsDofMapper mapper = mPatch.getMapper((T)1e-4);
40
41 // Get global coefficients of the multi patch. (i.e. without duplicates on the interfaces)
42 gsMatrix<T> globalCoefs(mapper.size(), 4); // 4 is the number of physical dimensions + 1 for weights, bext assumes 3D
43 globalCoefs.setZero();
44 globalCoefs.col(3).setOnes(); // Set weights to 1
45 index_t WFL = 0; // Weight Flag
46 // Loop over all patches
47 for (index_t p = 0; p != mPatch.nPatches(); p++)
48 {
49 for (index_t i=0; i != mPatch.patch(p).coefs().rows(); ++i) // For every control point
50 {
51 if( 0==WFL && mPatch.basis(p).isRational()) WFL = 1; // Set weight flag
52 globalCoefs.row(mapper.index(i,p)).leftCols(mPatch.geoDim()) = mPatch.patch(p).coefs().row(i);
53 if (mPatch.basis(p).isRational())
54 // Write the weight, assuming that both patches have the same weight on the interface
55 globalCoefs(mapper.index(i,p),3) = mPatch.basis(p).weights()(i);
56 }
57 }
58
59 std::ofstream file;
60 file.open(filename);
61
62 file << "B E X T 2.1\n"; // HEADER
63 std::map< std::array<size_t, 4>, internal::ElementBlock> ElementBlocks = mPatch.BezierOperator();
64
65 gsVector<index_t> NCVd, NCVCd; // Number of Coefficient Vectors per Block, Size of coefficient vectors (per block)
66 NCVd.setZero(ElementBlocks.size());
67 NCVCd.setZero(ElementBlocks.size());
68 size_t block=0;
69 for (auto const& pair : ElementBlocks) // For every element block
70 {
71 NCVCd[block] = pair.second.coefVectors.front().cols(); // Number of coefficient vector components ( == Bezier Basis Size)
72 for (auto const& bezCoefs : pair.second.coefVectors)
73 {
74 // GISMO_ASSERT(NCVCd == bezCoefs.rows(), "Dimension mismatch between Bezier Basis Size and CoefVector rows.");
75 NCVd[block] += bezCoefs.rows(); // Coefficient vectors total count.
76 }
77 block++;
78 }
79
80 // TEXT BLOCK 1 - PATCH
81 file << "#BLOCK 1 - PATCH==========================================================\n";
82 file << "# NN, NE, NCV, WFL, DIM\n";
83 size_t numBezElmnts=0;
84 for (auto const& pair : ElementBlocks) // For every element block
85 numBezElmnts += pair.second.numElements;
86 file << INT << globalCoefs.rows() << ","; // Number of initial control points
87 file << INT << numBezElmnts << ","; // Number of Bezier Elements
88 file << INT << NCVd.sum() << ","; // Number of Bezier Coefficient Vectors
89 file << INT << WFL << ","; // Flag for NURBS
90 file /* << INT << mPatch.domainDim() */ << "\n"; // Domain Dimension
91
92 // TEXT BLOCK 2 - NODES
93 file << "#BLOCK 2 - NODES==========================================================\n";
94 file << "# Xi, Yi, Zi, Wi\n";
95 for (index_t point=0; point!= globalCoefs.rows(); point++)
96 {
97 for (index_t c=0; c != globalCoefs.cols(); c++)
98 {
99 file << DOUBLE << globalCoefs(point,c);
100 if (c!= globalCoefs.cols()-1) file << ',';
101 else file << '\n';
102 }
103 }
104
105 // TEXT BLOCK 3 - ELEMENTS
106 file << "#BLOCK 3 - ELEMENTS=======================================================\n";
107 file << "# NEB\n";
108 file << INT << ElementBlocks.size() << std::endl; // NEB = Number of Element Blocks
109
110 file << "# ETYPE, NE, NNj, NCVj, PRj, PSj, PTj\n";
111 index_t NNj; // Number of original nodes associated with each Bezier element.
112 for (auto const& pair : ElementBlocks) // For every block
113 {
114 NNj = pair.first.at(0);
115 file << INT << 1 << ','; // ETYPE, FiXED for now
116
117 file << INT << pair.second.numElements << ','; // NE Numbers of Elements in each block
118 file << INT << pair.second.actives.front().rows() << ','; // NNj Number of Nodes for block j
119 file << INT << pair.second.coefVectors.front().rows() << ','; // NCVj Numebr of Coefficient Vectors for block j
120 file << INT << pair.second.PR << ','; // PRj Polynomial Degree 1st dir
121 file << INT << pair.second.PS << ','; // PSj 2nd dir
122 file << INT << 0 << std::endl; // PTj 3rd dir
123 }
124
125 index_t i = 1;
126 block=0;
127 for (auto const& pair : ElementBlocks) // For every block
128 {
129 file << "#" << std::setfill('-') << std::setw(24) << "Block"<<block<<"\n";
130 auto Ait = pair.second.actives.begin(); // Actives Iterator
131 auto Cit = pair.second.coefVectors.begin(); // Coefficients Iteratos
132
133 for(; Ait != pair.second.actives.end() && Cit != pair.second.coefVectors.end(); ++Ait, ++Cit)
134 {
135 GISMO_ASSERT(Ait->rows() == Cit->rows(), "Dimensions mismatch, the number of nodes ("<< Ait->rows() << ") and the rows of the CoefVectors ("<< Cit->rows() << ")must be equal.");
136 // Node ID line
137 file << "# Nk,\n";
138 for (index_t j=0; j != Ait->rows(); j++ )
139 { // Export Node IDs ( aka actives ), 10 per line
140 if (j!=0 && j%10 == 0)
141 {
142 file << std::endl;
143 }
144 file << INT << (*Ait)(j,0) + 1; // +1 because FORTRAN
145 if (j!=Ait->rows()-1 || j%10 == 9) file << ','; // Append comma, if not at the last entry of the line
146 }
147 file << std::endl;
148
149 // CVID line
150 file << "# CVIDl,\n";
151 for (index_t j=0; j != Cit->rows(); j++ ) // TODO: Not sure if cols or rows
152 { // Since at block 4 we will write all coefficient vector entries serially,
153 // the CVID are just an increasing integer. 10 per line.
154 if (j!=0 && j%10 == 0)
155 {
156 file << std::endl;
157 }
158 file << INT << i; // i starts from 1, Because FORTRAN
159 if (j!=Cit->rows()-1 && j%10 != 9) file << ','; // Append comma, if not at the last entry of the line
160 i++;
161 }
162 file << std::endl;
163 file << std::endl;
164 }
165 block++;
166 }
167
168
169
170
171 // TEXT BLOCK 4 - COEFFICIENT VECTORS
172 file << "#BLOCK 4 - COEFFICIENT VECTORS============================================\n";
173 file << "# NDCVB, NSCVB\n";
174 file << INT << ElementBlocks.size() << ','; // Number of Dense coef blocks (all are exported as dense)
175 file << INT << 0 << std::endl; // Number of sparse blocks ( None )
176
177 file << "# NCVd, NCVCd\n";
178 for (block = 0; block != ElementBlocks.size(); block++ ) // For every block
179 {
180 file << INT << NCVd[block] << ','; // Number of dense Bezier coefficient vector entries
181 file << INT << NCVCd[block] << std::endl; // Width of each entry ( i.e. number of bezier coefficents per entry)
182 }
183
184 block=0;
185 for (auto const& pair : ElementBlocks) // For every block
186 {
187 file << "#" << std::setfill('-') << std::setw(22) << "Block"<<block<<"\n";
188 file << "# CVCm, Bezier Coefficients\n";
189 auto Cit = pair.second.coefVectors.begin(); // Coefficient Vector Iterator
190 for(; Cit != pair.second.coefVectors.end(); ++Cit)
191 {
192 GISMO_ASSERT(Cit->cols() == (pair.second.PR+1)*(pair.second.PS+1)*(pair.second.PT+1),
193 "Dimensions mismatch, the columns of the Bezier Extraction Operator (Coefficient Vectors) "
194 "must be equal to the size of the Bezier Basis.");
195 GISMO_ASSERT(Cit->rows() == pair.first.at(0),
196 "Dimensions mismatch, the rows of the Bezier Extraction Operator (Coefficient Vectors) "
197 "must be equal to the number of the nodes (NNj) associated with this element.");
198 for ( index_t row=0; row!=Cit->rows(); row++) // For each column, the
199 {
200 for (index_t col=0; col!=Cit->cols(); col++)
201 { // Export the coefficients ( colWise ), 5 per line
202 if (col!=0 && col%5 == 0)
203 {
204 file << std::endl;
205 }
206 file << DOUBLE << (*Cit)(row, col) ;
207 if (col!=Cit->rows()-1 && col%5 != 4) file << ','; // Append comma, if not at the last entry of the line
208 }
209 file << std::endl;
210 }
211 }
212 block++;
213 }
214 file.close();
215}
