339  typename gsTHBSpline<d,T>::BoundaryGeometryType & result) const
340 {
341  GISMO_ASSERT(d-1>=0,"d must be greater or equal than 1");

348  if(d==1)

350  gsMatrix<T> val(1,1),point;
