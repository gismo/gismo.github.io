116// Row 0

118 R(0, 1) = rotVec.x() * rotVec.y() * (1 - cos_t) - rotVec.z() * sin_t;
119 R(0, 2) = rotVec.x() * rotVec.z() * (1 - cos_t) + rotVec.y() * sin_t;
120// Row 1

122 R(1, 1) = cos_t + rotVec.y() * rotVec.y() * (1 - cos_t);

127 R(2, 2) = cos_t + rotVec.z() * rotVec.z() * (1 - cos_t);
