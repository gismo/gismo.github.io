513  static void rref_impl(const gsEigen::MatrixBase<Derived>& Mat)
514  {
515  // todo: const T tol = 0

521  {
522  if (nc <= piv) return;
523  i = r;
524  while (0 == M(i, piv)) //~
525  {
526  ++i;
527  if (nr == i)
528  {
529  i = r;
530  ++piv;
531  if (nc == piv) return;
532  }
533  }
534
