23template<class T>
24void gsINSBlockAssembler<T>::initMembers()
25{
26 m_udofs = m_dofMappers.front().freeSize();
27 m_pdofs = m_dofMappers.back().freeSize();
28 m_pshift = m_tarDim * m_udofs;
29 m_dofs = m_tarDim * m_udofs + m_pdofs;
30
31 m_nonZerosPerColU = 1;
32 for (int i = 0; i < m_tarDim; i++)
33 m_nonZerosPerColU *= 2 * getBases().front().maxDegree(i) + 1;
34
35 m_nonZerosPerColP = 1;
36 for (int i = 0; i < m_tarDim; i++)
37 m_nonZerosPerColP *= 2 * getBases().back().maxDegree(i) + 1;
38
39 m_ddof[0].setZero(m_dofMappers.front().boundarySize(), m_tarDim);
40 m_ddof[1].setZero(m_dofMappers.back().boundarySize(), 1);
41
42 m_blockA.resize(m_udofs, m_udofs);
43
44 m_blockB.resize(m_tarDim);
45 m_blockMinusBT.resize(m_tarDim);
46 for (int s = 0; s < m_tarDim; ++s)
47 {
48 m_blockB[s].resize(m_pdofs, m_udofs);
49 m_blockMinusBT[s].resize(m_udofs, m_pdofs);
50 }
51
52 m_blockCT.resize(m_tarDim);
53 for (int s = 0; s < m_tarDim; ++s)
54 {
55 m_blockCT[s].resize(m_udofs, m_pdofs);
56 }
57
58 m_blockN.resize(m_udofs, m_udofs);
59 m_blockNpattern.resize(m_udofs, m_udofs);
60 m_blockM.resize(m_udofs, m_udofs);
61 m_blockAp.resize(m_pdofs, m_pdofs);
62 m_blockMp.resize(m_pdofs, m_pdofs);
63
64 m_rhsA.setZero(m_dofs, 1);
65 m_rhsB.setZero(m_dofs, 1);
66 m_rhsC.setZero(m_dofs, 1);
67 m_rhsN.setZero(m_dofs, 1);
68 m_rhsM.setZero(m_dofs, 1);
69 m_rhsAp.setZero(m_dofs, 1);
70 m_rhsMp.setZero(m_dofs, 1);
71 m_rhsF.setZero(m_dofs, 1);
72
73 m_solution.setZero(m_dofs, 1);
74
75 // if (getAssemblerOptions().intStrategy == iFace::dg)
76 // {
77 // m_blockPdg.resize(m_pdofs, m_pdofs);
78 // m_rhsPdg.setZero(m_dofs, 1);
79 // }

81 if (getAssemblerOptions().dirStrategy == dirichlet::elimination)
82 {
83 computeDirichletDofs(0, 0, m_ddof[0]);
84 computeDirichletDofs(1, 1, m_ddof[1]);
85 }
86
87 m_currentVelField = constructSolution(m_solution, 0);
88 m_oldTimeVelField = m_currentVelField;
89
90 m_bPCDbndPrepared = false;

93template<class T>
94void gsINSBlockAssembler<T>::constructSolution(const gsMatrix<T>& solVector, gsMultiPatch<T>& result, int unk) const
95{
96 GISMO_ASSERT(m_dofs == solVector.rows(), "Something went wrong, is solution vector valid?");
97
98 const gsDofMapper & mapper = m_dofMappers[unk];
99
100 result.clear(); // result is cleared first
101
102 const index_t dim = (unk == 0 ? m_tarDim : 1);
103 gsMatrix<T> coeffs;
104
105 // Point to the correct entries of the solution vector
106 gsAsConstMatrix<T> solV = (unk == 0 ?
107 gsAsConstMatrix<T>(solVector.data(), m_udofs, dim)
108 :
109 gsAsConstMatrix<T>(solVector.data() + m_pshift, m_pdofs, 1)
110 );
111
112 for (unsigned int p = 0; p < getPatches().nPatches(); ++p)
113 {
114 // Reconstruct solution coefficients on patch p
115 const int sz = getBases().at(unk).piece(p).size();
116 coeffs.resize(sz, dim);
117
118 for (index_t i = 0; i < sz; ++i)
119 {
120 if (mapper.is_free(i, p)) // DoF value is in the solVector
121 {
122 coeffs.row(i) = solV.row(mapper.index(i, p));
123 }
124 else // eliminated DoF: fill with Dirichlet data
125 {
126 coeffs.row(i) = m_ddof[unk].row(mapper.bindex(i, p));
127 }
128 }
129
130 result.addPatch(getBases().at(unk).piece(p).makeGeometry(coeffs));
131 }
132}

158template<class T>
159T gsINSBlockAssembler<T>::computeFlowRate(int patch, boxSide side, gsMatrix<T> solution) const
160{
161 T flowRate = 0;
162
163 gsField<T> solutionField = constructSolution(solution, 0); // velocity field
164
165 const gsGeometry<T>& geo = getPatches().patch(patch);
166 const gsBasis<T>& basis = getBases().at(0).basis(patch);
167
168 gsVector<int> numQuadNodes(m_tarDim);
169 const int dir = side.direction();
170 for (int i = 0; i < m_tarDim; ++i)
171 numQuadNodes[i] = (2 * basis.degree(i) + 1);
172 numQuadNodes[dir] = 1;
173
174 // Setup Quadrature
175 gsGaussRule<T> QuRule(numQuadNodes);
176
177 gsMatrix<T> quNodes; // Mapped nodes
178 gsVector<T> quWeights; // Mapped weights
179
180 // Initialize geometry evaluator
181 gsMapData<T> mapData;
182 mapData.flags = NEED_VALUE | NEED_OUTER_NORMAL;
183
184 typename gsBasis<T>::domainIter domIt = basis.makeDomainIterator(side);
185 for (; domIt->good(); domIt->next())
186 {
187 // Compute the quadrature rule on patch1
188 QuRule.mapTo(domIt->lowerCorner(), domIt->upperCorner(), quNodes, quWeights);
189
190 // Compute image of Gauss nodes under geometry mapping as well as Jacobians
191 mapData.points = quNodes;
192 geo.computeMap(mapData);
193
194 // Evaluate solution on element nodes
195 gsMatrix<T> solUVals = solutionField.value(quNodes, patch);
196
197 for (index_t k = 0; k < quWeights.rows(); ++k) // loop over quadrature nodes
198 {
199 // Compute the outer normal vector from patch1
200 gsVector<T> normal;
201 outerNormal(mapData, k, side, normal);
202
203 // the normal norm is equal to integral measure
204 flowRate += quWeights[k] * normal.dot(solUVals.col(k));
205 }
206 }
207 return flowRate;
208}

363template<class T>
364void gsINSBlockAssembler<T>::fillStokesSystem_into(gsSparseMatrix<T> & stokesMatrix, gsMatrix<T> & stokesRhs) const
365{
366 gsVector<int> nonZerosPerColumnVector;
367 nonZerosPerColumnVector.setZero(m_dofs);
368 for (int s = 0; s < m_tarDim; ++s)
369 for (int i = 0; i < m_udofs; i++)
370 {
371 nonZerosPerColumnVector(i + s * m_udofs) = m_blockA.col(i).nonZeros() + m_blockB[s].col(i).nonZeros();
372 }
373
374 for (int i = 0; i < m_pdofs; i++)
375 for (int s = 0; s < m_tarDim; ++s)
376 nonZerosPerColumnVector(i + m_pshift) += m_blockMinusBT[s].col(i).nonZeros();
377
378 // if (getAssemblerOptions().intStrategy == iFace::dg)
379 // {
380 // for (int i = 0; i < m_pdofs; i++)
381 // nonZerosPerColumnVector(i + m_pshift) += m_blockPdg.col(i).nonZeros();
382 // }
383
384 stokesMatrix.resize(m_dofs, m_dofs); // Clean matrix
385 stokesMatrix.reserve(nonZerosPerColumnVector);
386
387#pragma omp parallel for num_threads(m_numThreads)
388 for (index_t col = 0; col < m_udofs; ++col)
389 for (typename gsSparseMatrix<T>::InnerIterator it(m_blockA, col); it; ++it)
390 for (index_t s = 0; s < m_tarDim; ++s)
391 stokesMatrix.insert(it.row() + s*m_udofs, it.col() + s*m_udofs) = it.value();
392
393#pragma omp parallel for num_threads(m_numThreads)
394 for (index_t col = 0; col < m_pshift; ++col)

397
398#pragma omp parallel for num_threads(m_numThreads)
399 for (index_t col = 0; col < m_pdofs; ++col)
400 for (index_t s = 0; s < m_tarDim; ++s)
401 for (typename gsSparseMatrix<T>::InnerIterator it(m_blockMinusBT[s], col); it; ++it)
402 stokesMatrix.insert(it.row() + s*m_udofs, it.col() + m_pshift) = it.value();

404// if (getAssemblerOptions().intStrategy == iFace::dg)

409// stokesMatrix.insert(it.row() + m_pshift, it.col() + m_pshift) = it.value();
410// }
411
412 stokesMatrix.makeCompressed();

414 stokesRhs = m_rhsA + m_rhsB + m_rhsF;

416 // if (getAssemblerOptions().intStrategy == iFace::dg)

464template<class T>
465void gsINSBlockAssembler<T>::assembleApRobinBlock(std::vector<std::pair<int, boxSide> > bndPart)
466{
467 // matrix and rhs cleaning
468 m_blockRobin.resize(m_pdofs, m_pdofs);
469 gsMatrix<T> dummyRhs(m_dofs, 1);
470 dummyRhs.setZero();
471
472 m_blockRobin.reserve(gsVector<int>::Constant(m_pdofs, m_nonZerosPerColP));
473
474 gsINSBlockVisitorRobinPCD<T> visitor(m_dofMappers, m_viscosity);
475 this->blockSpecificSettings(&visitor);
476
477 gsMatrix<T> quNodes, physNodes;
478 gsVector<T> quWeights;
479 unsigned evFlags(0);
480 gsQuadRule<T> QuRule;
481 gsMapData<T> mapData;
482 typename gsBasis<T>::domainIter domIt;
483 memory::shared_ptr< gsBasisRefs<T> > basesPtr;
484
485 for (size_t sideIndex = 0; sideIndex < bndPart.size(); sideIndex++) // loop over inflow patch sides
486 {
487 int patch = bndPart[sideIndex].first;
488 boxSide side = bndPart[sideIndex].second;
489
490 basesPtr.reset(new gsBasisRefs<T>(getBases(), patch));
491
492 visitor.initialize(*basesPtr, patch, QuRule, evFlags, side);
493
494 mapData.flags = evFlags;
495
496 domIt = (*basesPtr)[1].makeDomainIterator(side);
497
498 for (; domIt->good(); domIt->next())
499 {
500 QuRule.mapTo(domIt->lowerCorner(), domIt->upperCorner(), quNodes, quWeights);
501 mapData.points = quNodes;
502 getPatches().patch(patch).computeMap(mapData);
503
504 visitor.evaluate(*basesPtr, mapData, quNodes);
505 visitor.assemble(*domIt, mapData, quWeights);
506 visitor.localToGlobal(m_blockRobin, dummyRhs);
507 }
508 }
509}

525template<class T>
526gsSparseMatrix<T> gsINSBlockAssembler<T>::getPressurePoissonMatrix(bool assemb, bool lumping)
527{
528 if (assemb)
529 {
530 if (!m_blockAp.nonZeros())
531 assemblePressurePoisson();
532
533 return m_blockAp;
534 }
535 else
536 {
537 gsSparseMatrix<T> velMinv(m_udofs, m_udofs);
538 diagInvMatrix_into(m_blockM, velMinv, 1, lumping);
539
540 gsSparseMatrix<T> Ap = m_blockB[0] * velMinv * m_blockB[0].transpose();
541 for (size_t i = 1; i < m_blockB.size(); i++)
542 Ap += m_blockB[i] * (velMinv * m_blockB[i].transpose());
543
544 return Ap;
545 }

585template<class T>
586void gsINSBlockAssembler<T>::findPressureBoundaryPartIDs(std::vector<std::pair<int, boxSide> > bndPart, std::vector<index_t>& idVector)
587{
588 const gsMultiBasis<T>& pBasis = getBases()[1];
589
590 gsMatrix<index_t> bnd;
591 idVector.clear();
592
593 for (std::vector<std::pair<int, boxSide> >::iterator it = bndPart.begin(); it != bndPart.end(); it++)
594 {
595 bnd = pBasis.piece(it->first).boundary(it->second);

603template<class T>
604void gsINSBlockAssembler<T>::applyPCDboundaryDirichletPart(gsSparseMatrix<T>& block, const std::vector<index_t>& bndPartIDs, bool scaleDiag, bool exclInflowPart)
605{
606 GISMO_ASSERT(m_bPCDbndPrepared, "Pressure boundary DOF IDs not prepared!");
607
608 gsSparseMatrix<T> mat(block.rows(), block.cols());
609 mat.setIdentity();
610
611 for (size_t i = 0; i < bndPartIDs.size(); i++)
612 mat.coeffRef(bndPartIDs[i], bndPartIDs[i]) = 0;
613
614 mat.prune(0,0);
615
616 block = mat * block * mat;
617
618 gsMatrix<T> diag = block.diagonal();
619
620 if (exclInflowPart)
621 {
622 for (size_t i = 0; i < m_presInIDs.size(); i++)
623 diag(m_presInIDs[i]) = 0;
624 }
625
626 real_t alpha = 1.0;
627
628 if(scaleDiag)
629 alpha = diag.sum() / diag.count(); // average of the diagonal entries not corresponding to bndPart (+ inflow)
630
631 block.reserve(gsVector<int>::Constant(block.cols(), 1));
632
633 for (size_t i = 0; i < bndPartIDs.size(); i++)
634 block.coeffRef(bndPartIDs[i], bndPartIDs[i]) = alpha;
635
636 block.makeCompressed();
637}
