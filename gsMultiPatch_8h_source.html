338  void repairInterfaces()
339  {
340  std::vector< boundaryInterface > bivec = interfaces();
341  size_t kmax = 2*bivec.size();
342  size_t k = 0;
343  bool sthChanged = false;
344  bool change = false;
345  do
346  {
347  sthChanged = false;
348  for( size_t i = 0; i < bivec.size(); i++ )
349  {
350  change = repairInterface( bivec[i] );
351  sthChanged = sthChanged || change;
352  }
353  k++; // just to be sure this cannot go on infinitely
354  }
355  while( sthChanged && k <= kmax );
356  }
