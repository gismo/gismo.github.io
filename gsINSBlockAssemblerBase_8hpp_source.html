151template<class T>
152void gsINSBlockAssemblerBase<T>::computeDirichletDofs(const int unk, const int basisID, gsMatrix<T>& ddofVector)
153{
154 GISMO_ASSERT(ddofVector.rows() == m_dofMappers[basisID].boundarySize(), "Dirichlet DOF vector has wrong size.");
155
156 const gsDofMapper & mapper = m_dofMappers[basisID];
157 const gsMultiBasis<T> & mbasis = getBases().at(basisID);
158
159 switch (getAssemblerOptions().dirValues)
160 {
161 case dirichlet::homogeneous:
162 ddofVector.setZero();
163 break;
164 case dirichlet::interpolation:
165 computeDirichletDofsIntpl(unk, mapper, mbasis, ddofVector);
166 break;
167 case dirichlet::l2Projection:
168 computeDirichletDofsL2Proj(unk, mapper, mbasis, ddofVector);
169 break;
170 default:
171 GISMO_ERROR("Something went wrong with Dirichlet values.");
172 }
173
174 // Corner values
175 for (typename gsBoundaryConditions<T>::const_citerator
176 it = getBCs().cornerBegin();
177 it != getBCs().cornerEnd(); ++it)
178 {
179 if (it->unknown == unk)
180 {
181 const int i = mbasis[it->patch].functionAtCorner(it->corner);
182 const int ii = mapper.bindex(i, it->patch);
183 ddofVector.row(ii).setConstant(it->value);
184 }
185 else
186 continue;
187 }
188}

191template<class T>
192void gsINSBlockAssemblerBase<T>::computeDirichletDofsIntpl(const int unk, const gsDofMapper & mapper, const gsMultiBasis<T> & mbasis, gsMatrix<T>& ddofVector)
193{
194 for (typename gsBoundaryConditions<T>::const_iterator
195 it = getBCs().dirichletBegin();
196 it != getBCs().dirichletEnd(); ++it)
197 {
198 if (it->unknown() != unk)
199 continue;
200
201 const int patchIdx = it->patch();
202 const gsBasis<T> & basis = mbasis.piece(patchIdx);
203
204 // Get dofs on this boundary
205 gsMatrix<index_t> boundary = basis.boundary(it->side());
206
207 // If the condition is homogeneous then fill with zeros
208 if (it->isHomogeneous())
209 {
210 for (index_t i = 0; i != boundary.size(); ++i)
211 {
212 const int ii = mapper.bindex((boundary)(i), it->patch());
213 ddofVector.row(ii).setZero();
214 }
215 continue;
216 }
217
218 // Get the side information
219 int dir = it->side().direction();
220 index_t param = (it->side().parameter() ? 1 : 0);
221
222 // Compute grid of points on the face ("face anchors")
223 std::vector< gsVector<T> > rr;
224 rr.reserve(getPatches().parDim());
225
226 for (int i = 0; i < getPatches().parDim(); ++i)
227 {
228 if (i == dir)
229 {
230 gsVector<T> b(1);
231 b[0] = (basis.component(i).support()) (0, param);
232 rr.push_back(b);
233 }
234 else
235 {
236 rr.push_back(basis.component(i).anchors().transpose());
237 }
238 }
239
240 GISMO_ASSERT(it->function()->targetDim() == ddofVector.cols(),
241 "Given Dirichlet boundary function does not match problem dimension."
242 << it->function()->targetDim() << " != " << ddofVector.cols() << "\n");
243
244 // Compute dirichlet values
245 gsMatrix<T> pts = getPatches().patch(it->patch()).eval(gsPointGrid<T>(rr));
246 gsMatrix<T> fpts;
247 if (!it->parametric())
248 {
249 fpts = it->function()->eval(pts);
250 }
251// else if (it->parametric() && m_params.settings().isIgaDirichletGeometry())
252// {
253// gsMatrix<T> parPts;
254// getIgaBCGeom().invertPoints(pts, parPts);
255// fpts = it->function()->eval(parPts);
256// }
257 else
258 fpts = it->function()->eval(gsPointGrid<T>(rr));
259
260 // Interpolate dirichlet boundary
261 typename gsBasis<T>::uPtr h = basis.boundaryBasis(it->side());
262 typename gsGeometry<T>::uPtr geo = h->interpolateAtAnchors(fpts);
263 const gsMatrix<T> & dVals = geo->coefs();
264
265 // Save corresponding boundary dofs
266 for (index_t i = 0; i != boundary.size(); ++i)
267 {
268 const int ii = mapper.bindex(boundary.at(i), it->patch());
269 ddofVector.row(ii) = dVals.row(i);
270 }
271 }
272}

275template<class T>
276void gsINSBlockAssemblerBase<T>::computeDirichletDofsL2Proj(const int unk, const gsDofMapper & mapper, const gsMultiBasis<T> & mbasis, gsMatrix<T>& ddofVector)
277{
278 // Set up matrix, right-hand-side and solution vector/matrix for
279 // the L2-projection
280 gsSparseEntries<T> projMatEntries;
281 gsMatrix<T> globProjRhs;
282 globProjRhs.setZero(ddofVector.rows(), ddofVector.cols());
283
284 // Temporaries
285 gsVector<T> quWeights;
286 gsMatrix<T> rhsVals;
287 gsMatrix<index_t> globIdxAct;
288 gsMatrix<T> basisVals;
289 gsMapData<T> mapData(NEED_MEASURE);
290
291 // Iterate over all patch-sides with Dirichlet-boundary conditions
292 for (typename gsBoundaryConditions<T>::const_iterator
293 it = getBCs().dirichletBegin();
294 it != getBCs().dirichletEnd(); ++it)
295 {
296 if (it->unknown() != unk)
297 continue;
298
299 if (it->isHomogeneous())
300 continue;
301
302 GISMO_ASSERT(it->function()->targetDim() == ddofVector.cols(),
303 "Given Dirichlet boundary function does not match problem dimension."
304 << it->function()->targetDim() << " != " << ddofVector.cols() << "\n");
305
306 const int patchIdx = it->patch();
307 const gsBasis<T> & basis = mbasis.piece(patchIdx);
308 const gsGeometry<T> & patch = getPatches()[patchIdx];
309
310 // Set up quadrature to degree+1 Gauss points per direction,
311 // all lying on it->side() except from the direction which
312 // is NOT along the element
313
314 gsGaussRule<T> bdQuRule(basis, 1.0, 1, it->side().direction());
315
316 // Create the iterator along the given part boundary.
317 typename gsBasis<T>::domainIter bdryIter = basis.makeDomainIterator(it->side());
318
319 for (; bdryIter->good(); bdryIter->next())
320 {
321 bdQuRule.mapTo(bdryIter->lowerCorner(), bdryIter->upperCorner(),
322 mapData.points, quWeights);
323
324 patch.computeMap(mapData);
325
326 // the values of the boundary condition are stored
327 // to rhsVals. Here, "rhs" refers to the right-hand-side
328 // of the L2-projection, not of the PDE.
329 rhsVals = it->function()->eval(getPatches()[patchIdx].eval(mapData.points));
330
331 basis.eval_into(mapData.points, basisVals);
332
333 // active basis (first line) functions/DOFs:
334 basis.active_into(mapData.points.col(0), globIdxAct);
335 mapper.localToGlobal(globIdxAct, patchIdx, globIdxAct);
336
337 // eltBdryFcts stores the row in basisVals/globIdxAct, i.e.,
338 // something like a "element-wise index"
339 std::vector<index_t> eltBdryFcts;
340 eltBdryFcts.reserve(mapper.boundarySize());
341 for (index_t i = 0; i < globIdxAct.rows(); i++)
342 if (mapper.is_boundary_index(globIdxAct(i, 0)))
343 eltBdryFcts.push_back(i);
344
345 // Do the actual assembly:
346 for (index_t k = 0; k < mapData.points.cols(); k++)
347 {
348 const T weight_k = quWeights[k] * mapData.measure(k);
349
350 // Only run through the active boundary functions on the element:
351 for (size_t i0 = 0; i0 < eltBdryFcts.size(); i0++)
352 {
353 // Each active boundary function/DOF in eltBdryFcts has...
354 // ...the above-mentioned "element-wise index"
355 const index_t i = eltBdryFcts[i0];
356 // ...the boundary index.
357 const index_t ii = mapper.global_to_bindex(globIdxAct(i));
358
359 for (size_t j0 = 0; j0 < eltBdryFcts.size(); j0++)
360 {
361 const index_t j = eltBdryFcts[j0];
362 const index_t jj = mapper.global_to_bindex(globIdxAct(j));
363
364 // Use the "element-wise index" to get the needed
365 // function value.
366 // Use the boundary index to put the value in the proper
367 // place in the global projection matrix.
368 projMatEntries.add(ii, jj, weight_k * basisVals(i, k) * basisVals(j, k));
369 } // for j
370
371 globProjRhs.row(ii) += weight_k * basisVals(i, k) * rhsVals.col(k).transpose();
372
373 } // for i
374 } // for k
375 } // bdryIter
376 } // boundaryConditions-Iterator
377
378 gsSparseMatrix<T> globProjMat(mapper.boundarySize(), mapper.boundarySize());
379 globProjMat.setFrom(projMatEntries);
380 globProjMat.makeCompressed();
381
382 // Solve the linear system:
383 // The position in the solution vector already corresponds to the
384 // numbering by the boundary index. Hence, we can simply take them
385 // for the values of the eliminated Dirichlet DOFs.
386 typename gsSparseSolver<T>::CGDiagonal solver;
387 ddofVector = solver.compute(globProjMat).solve(globProjRhs);
388}

391template<class T>
392template<class ElementBlockVisitor>
393void gsINSBlockAssemblerBase<T>::assembleBlock(gsSparseMatrix<T> & matrixBlock, gsMatrix<T> & rhs, const int nonZerosPerCol)
394{
395 std::vector< gsSparseMatrix<T> *> threadBlockPtr;
396 threadBlockPtr.resize(m_numThreads);
397 threadBlockPtr[0] = &matrixBlock;
398 for (int t = 1; t < m_numThreads; ++t) {
399 threadBlockPtr[t] = new gsSparseMatrix<T>(matrixBlock.rows(), matrixBlock.cols());
400 threadBlockPtr[t]->reserve(gsVector<int>::Constant(matrixBlock.cols(), nonZerosPerCol));
401 }
402
403 std::vector< gsMatrix<T> * > threadRhsPtr;
404 threadRhsPtr.resize(m_numThreads);
405 threadRhsPtr[0] = &rhs;
406 for (int t = 1; t < m_numThreads; ++t) {
407 threadRhsPtr[t] = new gsMatrix<T>(rhs.rows(), rhs.cols());
408 threadRhsPtr[t]->setZero();
409 }
410
411 #pragma omp parallel num_threads(m_numThreads)
412 {
413 #ifdef _OPENMP
414 const int threadNumber(omp_get_thread_num());
415 #else
416 const int threadNumber = 0;
417 #endif
418
419 // deep copies
420 const gsMultiPatch<T> patches(getPatches());
421 std::vector< gsMultiBasis<T> > bases(getBases());
422 std::vector< gsDofMapper > dofMappers(m_dofMappers);
423 const std::vector<gsMatrix<T> > ddof(m_ddof);
424 const index_t totalElementsNumber(m_elementList.size());
425 const std::vector< std::pair<index_t, index_t> > elementList(m_elementList);
426
427 gsQuadRule<T> QuRule;
428 gsMatrix<T> quNodes;
429 gsVector<T> quWeights;
430 unsigned evFlags(0);
431 gsMapData<T> mapData;
432 memory::shared_ptr< gsBasisRefs<T> > basesPtr;
433 typename gsBasis<T>::domainIter domIt;
434
435 gsSparseMatrix<T> * blockPtr = threadBlockPtr[threadNumber];
436 gsMatrix<T> * rhsPtr = threadRhsPtr[threadNumber];
437
438 index_t patchIndex = -1;
439 index_t elementIndex = -1;
440
441 // Parallel assembly procedure.
442 #pragma omp for
443 for (index_t nElementGlobal = 0; nElementGlobal < totalElementsNumber; ++nElementGlobal) {
444
445 ElementBlockVisitor blockVisitor(dofMappers, m_viscosity);
446
447 blockSpecificSettings(&blockVisitor);
448
449 if (patchIndex != elementList[nElementGlobal].first)
450 {
451 patchIndex = elementList[nElementGlobal].first;
452
453 basesPtr.reset(new gsBasisRefs<T>(bases, patchIndex));
454 domIt = basesPtr->front().makeDomainIterator(boundary::none);
455 elementIndex = 0;
456 }
457
458 blockVisitor.initialize(*basesPtr, patchIndex, QuRule, evFlags);
459
460 mapData.flags = evFlags;
461
462 if (elementIndex != elementList[nElementGlobal].second)
463 {
464 while ((elementIndex < elementList[nElementGlobal].second) && (domIt->good())) {
465 domIt->next();
466 ++elementIndex;
467 }
468 }
469
470 QuRule.mapTo(domIt->lowerCorner(), domIt->upperCorner(), quNodes, quWeights);
471
472 mapData.points = quNodes;
473 patches[patchIndex].computeMap(mapData);
474
475 blockVisitor.evaluate(*basesPtr, quNodes);//, *domIt);
476
477 blockVisitor.assemble(mapData, quWeights);
478
479 blockVisitor.localToGlobal(ddof, *blockPtr, *rhsPtr);
480
481 domIt->next();
482 ++elementIndex;
483 }
484 }
485
486 // Summing up block matrices and rhs from all threads
487 // into threadBlockPtr[0], threadRhsPtr[0] which are
488 // pointers to the input variables & matrixBlock, & rhs
489 for (int levelSkip = 1; levelSkip < m_numThreads; levelSkip *= 2) {
490 #pragma omp parallel for num_threads(m_numThreads)
491 for (int i = 0; i < m_numThreads; i += 2 * levelSkip) {
492 if (i + levelSkip < m_numThreads) {
493 *threadBlockPtr[i] += *threadBlockPtr[i + levelSkip]; //matrix loses allocated space!!
494 *threadRhsPtr[i] += *threadRhsPtr[i + levelSkip];
495 }
496 }
497 }
498
499 for (int t = 1; t < m_numThreads; ++t) {
500 delete threadBlockPtr[t];
501 delete threadRhsPtr[t];
502 }
503}

506template<class T>
507template<class RhsVisitor>
508void gsINSBlockAssemblerBase<T>::assembleRhs(gsMatrix<T> & rhs)
509{
510 std::vector< gsMatrix<T> * > threadRhsPtr;
511 threadRhsPtr.resize(m_numThreads);
512 threadRhsPtr[0] = &rhs;
513 for (int t = 1; t < m_numThreads; ++t) {
514 threadRhsPtr[t] = new gsMatrix<T>(rhs.rows(), rhs.cols());
515 threadRhsPtr[t]->setZero();
516 }
517
518 #pragma omp parallel num_threads(m_numThreads)
519 {
520 #ifdef _OPENMP
521 const int threadNumber(omp_get_thread_num());
522 #else
523 const int threadNumber = 0;
524 #endif
525
526 // deep copies
527 const gsMultiPatch<T> patches(getPatches());
528 std::vector< gsMultiBasis<T> > bases(getBases());
529 std::vector< gsDofMapper > dofMappers(m_dofMappers);
530 const index_t totalElementsNumber(m_elementList.size());
531 const std::vector< std::pair<index_t, index_t> > elementList(m_elementList);
532
533 gsQuadRule<T> QuRule;
534 gsMatrix<T> quNodes;
535 gsVector<T> quWeights;
536 unsigned evFlags(0);
537 gsMapData<T> mapData;
538 memory::shared_ptr< gsBasisRefs<T> > basesPtr;
539 typename gsBasis<T>::domainIter domIt;
540
541 gsMatrix<T> * rhsPtr = threadRhsPtr[threadNumber];
542
543 index_t patchIndex = -1;
544 index_t elementIndex = -1;
545
546 // Parallel assembly procedure.
547 #pragma omp for
548 for (index_t nElementGlobal = 0; nElementGlobal < totalElementsNumber; ++nElementGlobal) {
549
550 RhsVisitor rhsVisitor(dofMappers, m_viscosity);
551
552 rhsSpecificSettings(&rhsVisitor);
553
554 if (patchIndex != elementList[nElementGlobal].first) {
555 patchIndex = elementList[nElementGlobal].first;
556
557 basesPtr.reset(new gsBasisRefs<T>(bases, patchIndex));
558 domIt = basesPtr->front().makeDomainIterator(boundary::none);
559 elementIndex = 0;
560 }
561
562 rhsVisitor.initialize(*basesPtr, patchIndex, QuRule, evFlags);
563
564 mapData.flags = evFlags;
565
566 if (elementIndex != elementList[nElementGlobal].second) {
567 while ((elementIndex < elementList[nElementGlobal].second) && (domIt->good())) {
568 domIt->next();
569 ++elementIndex;
570 }
571 }
572
573 QuRule.mapTo(domIt->lowerCorner(), domIt->upperCorner(), quNodes, quWeights);
574
575 mapData.points = quNodes;
576 patches[patchIndex].computeMap(mapData);
577
578 rhsVisitor.evaluate(*basesPtr, mapData, quNodes);
579
580 rhsVisitor.assemble(mapData, quWeights);
581
582 rhsVisitor.localToGlobal(*rhsPtr);
583
584 domIt->next();
585 ++elementIndex;
586 }
587
588 }
589
590 // Summing up rhs from all threads
591 // into threadRhsPtr[0] which is a
592 // pointer to the input variable & rhs
593 for (int t = 1; t < m_numThreads; ++t)
594 *threadRhsPtr[0] += *threadRhsPtr[t];
595
596 for (int t = 1; t < m_numThreads; ++t) {
597 delete threadRhsPtr[t];
598 }
599}
