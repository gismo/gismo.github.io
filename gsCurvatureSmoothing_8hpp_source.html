532  for (index_t i=0;i<num_rows;i++){
533  m_A(i,(num_rows+i-1)%num_rows)=m_smooth1;
534  m_A(i,(i+1)%num_rows)=m_smooth1;

545  m_coefs=m_A*m_coefs;
