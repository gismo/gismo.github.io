75 {
76 gsMatrix<index_t> ind = this->basis().boundary(s); // get indices of the boundary DOF
77 gsMatrix<T> coeffs (ind.size(), m_multiPatch->geoDim()); // create matrix for boundary coefficients
78
79 const gsMatrix<T> & initialCoefs = m_multiPatch->patch(m_patchIndex).coefs();
80 const gsDofMapper & map = m_solution.mapper();
81
82
83 for (index_t i=0; i != ind.size(); i++ ) // For every control point of the boundary
84 {
85 for (index_t component=0; component!=m_multiPatch->geoDim(); component++)
86 {
87 const index_t ii = map.index(ind(i), m_patchIndex, component);
88 if ( map.is_free_index(ii) ) // DoF value is in the solVector
89 {
90 coeffs(i,component) = initialCoefs( ind(i,0), component ) + m_solution.coefs().at( ii );
91 }
92 else
93 {
94 coeffs(i,component) = initialCoefs( ind(i,0), component ) + m_solution.fixedPart().at( map.global_to_bindex(ii) );
95 }
96 }
97 }

99 typename gsBasis<T>::uPtr Bs = this->basis().boundaryBasis(s); // Basis for boundary side s
100 return Bs->makeGeometry( give(coeffs) );

139 void compute(const gsMatrix<T> & in, gsFuncData<T> & out) const
140 {
141 const unsigned flags = out.flags | NEED_ACTIVE;
142 const index_t numPt = in.cols();
143 const index_t numCo = targetDim();
144
145 gsFuncData<T> tmp(flags);
146 this->basis().compute(in, tmp);
147
148 out.values.resize(out.maxDeriv()+1);
149 out.dim.first = tmp.dim.first;
150 out.dim.second = numCo;
151 if ( flags & SAME_ELEMENT )
152 {
153 gsMatrix<T> coefM;
154 extractRows(tmp.active(0),coefM);
155
156 if (flags & NEED_VALUE)
157 out.values[0].noalias()=coefM.transpose()*tmp.values[0];
158 if (flags & NEED_DERIV)
159 {
160 const index_t derS = tmp.derivSize();
161 out.values[1].resize(derS*numCo,numPt);
162 for (index_t p=0; p< numPt; ++p)
163 out.values[1].reshapeCol(p, derS, numCo).noalias() = tmp.deriv(p)*coefM;
164 }
165 if (flags & NEED_DERIV2)
166 {
167 const index_t derS = tmp.deriv2Size();
168 out.values[2].resize(derS*numCo,numPt);
169 for (index_t p=0; p< numPt; ++p)
170 out.values[2].reshapeCol(p, derS, numCo).noalias() = tmp.deriv2(p)*coefM;
171 }
172 if (flags & NEED_ACTIVE)
173 this->active_into(in.col(0), out.actives);
174 }
175 else
176 {
177 gsMatrix<T> coefM;
178 const index_t derS = tmp.derivSize();
179 const index_t der2S = tmp.deriv2Size();
180
181 if (flags & NEED_VALUE) out.values[0].resize(numCo,numPt);
182 if (flags & NEED_DERIV) out.values[1].resize(numCo*derS,numPt);
183 if (flags & NEED_DERIV2) out.values[2].resize(numCo*der2S,numPt);
184 if (flags & NEED_ACTIVE) this->active_into(in, out.actives);
185
186 for (index_t p=0; p<numPt;++p)
187 {
188
189 extractRows(tmp.active(p), coefM);
190
191 if (flags & NEED_VALUE)
192 out.values[0].reshapeCol(p,1,numCo).noalias() = tmp.eval(p)*coefM;
193 if (flags & NEED_DERIV)
194 out.values[1].reshapeCol(p, derS, numCo).noalias() = tmp.deriv(p)*coefM;
195 if (flags & NEED_DERIV2)
196 out.values[2].reshapeCol(p, der2S, numCo).noalias() = tmp.deriv2(p)*coefM;
197 }

201 void linearCombination_into(const gsMatrix<>& u, const gsMatrix<T> & basisValues, gsMatrix<>& result) const
202 {
203 const gsMatrix<index_t> active = m_multiPatch->patch(m_patchIndex).basis().active(u);
204
205 const gsDofMapper & map = m_solution.mapper();
206 GISMO_ASSERT(m_solution.coefs().size()==map.freeSize(), "The solution vector has wrong dimensions: "<<m_solution.coefs().size()<<" != "<<map.freeSize());
207
208 const gsMatrix<T> & initialCoefs = m_multiPatch->patch(m_patchIndex).coefs();
209
210 const index_t numPts = basisValues.cols() ;
211 const index_t tarDim = m_solution.dim() ;
212 const index_t stride = basisValues.rows() / active.rows();
213
214 GISMO_ASSERT( active.rows() * stride == basisValues.rows(),
215 "Number of values "<<basisValues.rows()<< " and active "<<active.rows()<<" does not fit together");
216
217 result.resize( tarDim * stride, numPts );
218 result.setZero();
219
220 for ( index_t pt = 0; pt < numPts; ++pt ) // For pt, i.e., for every column of u
221 {
222 for ( index_t i = 0; i < active.rows(); ++i ) // for all nonzero basis functions
223 for ( index_t c = 0; c < tarDim; ++c ) // for all components of the geometry
224 {
225 // result.block( stride * c, pt, stride, 1).noalias() +=
226 // coefs( active(i,pt), c) * values.block( stride * i, pt, stride, 1);
227 const index_t ii = map.index(active(i, pt), m_patchIndex, c);
228 if ( map.is_free_index(ii) ) // DoF value is in the solVector
229 result.block( stride * c, pt, stride, 1).noalias() += ( m_solution.coefs().at(ii) + initialCoefs( active(i,pt), c) ) * basisValues.block( stride * i, pt, stride, 1);
230 else
231 result.block( stride * c, pt, stride, 1).noalias() += ( m_solution.fixedPart().at( map.global_to_bindex(ii) ) + initialCoefs( active(i,pt), c) ) * basisValues.block( stride * i, pt, stride, 1);
232 }
233 }
