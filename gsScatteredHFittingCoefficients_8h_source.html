65 const gsMatrix<>& center,

68{

85 const gsMatrix<>& center,

91 {
92 const gsMatrix<> point = parameters.col(i);
93
94 if ((point - center).squaredNorm() <= radius * radius)
95 {
96 indices.push_back(i);

113int maximum_enlarging_factor(const gsTHBSplineBasis<2>& basis,
114 const index_t basis_fun_index)
115{
116 // TO DO, compute this for each level in advance
117
118 // real_t diagonal_of_element = 0;
119
120 int level = basis.levelOf(basis_fun_index);
121 const gsTensorBSplineBasis<2>& ten_basis = *basis.getBases()[level];
122 const int degree = ten_basis.maxDegree();
123 const gsMatrix<> support = basis.support(basis_fun_index);
124 const real_t radius = (support.col(0) - support.col(1)).norm() / 2;
125
126 const gsKnotVector<>& kv1 = ten_basis.knots(0);
127 const gsKnotVector<>& kv2 = ten_basis.knots(1);
128
129 const std::vector<real_t> breaks1 = kv1.breaks();
130 const std::vector<real_t> breaks2 = kv2.breaks();
131
132 const real_t cell_side_x = maximum_interval(breaks1);
133 const real_t cell_side_y = maximum_interval(breaks2);
134
135 const real_t cell_side_x_square = cell_side_x * cell_side_x;
136 const real_t cell_side_y_square = cell_side_y * cell_side_y;
137
138 const real_t cell_diag = math::sqrt(cell_side_x_square + cell_side_y_square);
139
140 // delta^{ell-1}
141 // look at paper Cesare Bracco: adaptive scattered data fitting...
142 const real_t delta_ell_1 = cell_diag * (degree + 1);
143
144 const int K_ell = cast<real_t,index_t>(math::ceil(2 * delta_ell_1 / radius) + 1);
145
146 return K_ell;
147}

149void get_data_for_least_square_approx(const gsTHBSplineBasis<2>& basis,
150 const gsMatrix<>& parameters,
151 const int basis_fun_index,
152 int& degreeLS,
153 std::vector<index_t>& indicesLS,
154 bool circle)
155{
156 gsMatrix<> support = basis.support(basis_fun_index);
157 gsMatrix<> center = get_center(support);
158 real_t radius = (support.col(0) - support.col(1)).norm() / 2;
159
160 real_t u_diff = std::abs(support(0,0) - support(0,1)) / 2;
161 real_t v_diff = std::abs(support(1,0) - support(1,1)) / 2;
162
163 const int degree = basis.maxDegree();
164
165 const int K = maximum_enlarging_factor(basis, basis_fun_index);
166
167 for (int k = 1; k <= K; k++)
168 {
169 // std::cout << " enlarging k = " << k << std::endl;
170
171 if(circle)
172 indicesLS = get_points_inside_circle(parameters, center, radius * k);
173 else
174 indicesLS = get_points_inside_rectangle(parameters, center, u_diff * k, v_diff * k);
175
176 const int num_LS_points = static_cast<index_t>(indicesLS.size());
177
178 if (num_LS_points == 0)
179 {
180 continue;
181 }
182 else
183 {
184 int d = degree;
185 while ( num_LS_points < (d + 1) * (d + 2) / 2)
186 {
187 d -= 1;
188 }
189 degreeLS = d;
190 break;
191 }
192 }
193}

196void assemble_LS_system(const gsTHBSplineBasis<2>& basis,
197 const int basis_fun_index,
198 const std::vector<index_t>& indices,
199 const gsMatrix<>& parameters,
200 const gsMatrix<>& points,
201 const int degree,
202 gsMatrix<>& A,
203 gsMatrix<>& B)
204{
205 const int dimension = (degree + 1) * (degree + 2) / 2;
206 A.resize(dimension, indices.size());
207 B.resize(indices.size(), points.rows());
208
209 const gsMatrix<> support = basis.support(basis_fun_index);
210 TransformationToUnitSquare transformation(support);
211
212 // const real_t dx = support(0, 1) - support(0, 0);
213 // const real_t dy = support(1, 1) - support(1, 0);
214 // const real_t origin_x = support(0, 0);
215 // const real_t origin_y = support(1, 0);
216
217 for (size_t i = 0; i != indices.size(); i++)
218 {
219 const int index = indices[i];
220 real_t xx, yy;
221 const real_t x = parameters(0, index);
222 const real_t y = parameters(1, index);
223 transformation.transform(x, y, xx, yy);
224 // const real_t xx = (x - origin_x) / dx;
225 // const real_t yy = (y - origin_y) / dy;
226
227 int j = 0;
228 for (int exp_x = 0; exp_x <= degree; exp_x++)
229 {
230 for (int exp_y = 0; exp_y <= degree - exp_x; exp_y++)
231 {
232 A(j, i) = math::pow(xx, exp_x) * math::pow(yy, exp_y);
233 j++;
234 }
235 }
236 B.row(i) = points.col(index).transpose();
237 }
238}

274gsMatrix<> generate_points(const gsTHBSplineBasis<2>& basis,
275 const int basis_fun_index)
276{
277 const int level = basis.levelOf(basis_fun_index);
278 const gsTensorBSplineBasis<2>& ten_basis = *basis.getBases()[level];
279 const int degree = ten_basis.maxDegree();
280
281 const gsMatrix<> support = basis.support(basis_fun_index);
282
283 const real_t factor = 1.0 / (2 * (degree + 1));
284 const gsMatrix<> diff = factor * (support.col(1) - support.col(0));
285
286 const int n = degree + 1;
287
288 gsVector<unsigned> num_points(2);
289 num_points << n, n;
290
291 gsVector<> start = support.col(0);
292 start(0) += diff(0, 0);
293 start(1) += diff(1, 0);
294
295 gsVector<> end = support.col(1);
296 end(0) -= diff(0, 0);
297 end(1) -= diff(1, 0);
298
299 gsMatrix<> points = gsPointGrid(start, end, num_points);
300
301 // std::cout << "support: \n" << support << "\n"
302 // << "points:\n" << points << "\n"
303 // << "size: " << points.rows() << " x " << points.cols() << std::endl;
304
305 return points;
306}

373gsMatrix<> assemble_system_coefsLS_to_basis_fun(const gsTHBSplineBasis<2>& basis,
374 const int basis_fun_index,
375 gsMatrix<>& gridPoints,
376 int& basis_fun_tensor_index)
377{
378 const int level = basis.levelOf(basis_fun_index);
379 const gsTensorBSplineBasis<2>& ten_basis = *basis.getBases()[level];
380 const unsigned tensor_index = basis.flatTensorIndexOf(basis_fun_index);
381
382 gsMatrix<> tmpPoints = generate_points(basis, basis_fun_index);
383
384 gsMatrix<index_t> indices;
385 ten_basis.active_into(tmpPoints, indices);
386
387 std::set<index_t> set_indices = matrix_to_set(indices);
388 std::map<index_t, index_t> basis_to_cols = compute_ordering(set_indices);
389
390 gridPoints = computeGridPoints(ten_basis, basis.support(basis_fun_index),
391 set_indices);
392
393 gsMatrix<> eval;
394 ten_basis.eval_into(gridPoints, eval);
395 ten_basis.active_into(gridPoints, indices);
396
397 gsMatrix<> A(gridPoints.cols(), set_indices.size());
398 A.setZero();
399
400 for (int point_index = 0; point_index != eval.cols(); point_index++)
401 {
402 for (int index = 0; index != eval.rows(); index++)
403 {
404 const real_t basis_value = eval(index, point_index);
405 const unsigned basis_index = indices(index, point_index);
406 if (basis_to_cols.find(basis_index) != basis_to_cols.end())
407 {
408 const int col = basis_to_cols[basis_index];
409 A(point_index, col) = basis_value;
410 }
411 }
412 }
413
414 // std::for_each(set_indices.begin(), set_indices.end(), print);
415 // std::cout << std::endl;
416 // std::cout << "tensor_index:" << tensor_index << "\n"
417 // << "col: " << basis_to_cols[tensor_index] << std::endl;
418
419 basis_fun_tensor_index = basis_to_cols[tensor_index];
420 return A;
421}

424gsMatrix<> polynomial_basis_evaluation(const int degreeLS,
425 const gsMatrix<>& gridPoints,
426 const gsMatrix<>& support)
427{
428 const int dimension = (degreeLS + 1) * (degreeLS + 2) / 2;
429 gsMatrix<> P(gridPoints.cols(), dimension);
430 TransformationToUnitSquare transformation(support);
431
432 for (int point_index = 0; point_index != gridPoints.cols(); point_index++)
433 {
434 int j = 0;
435 real_t xx, yy;
436 const real_t x = gridPoints(0, point_index);
437 const real_t y = gridPoints(1, point_index);
438 transformation.transform(x, y, xx, yy);
439
440 for (int exp_x = 0; exp_x <= degreeLS; exp_x++)
441 {
442 for (int exp_y = 0; exp_y <= degreeLS - exp_x; exp_y++)
443 {
444 P(point_index, j) = math::pow(xx, exp_x) * math::pow(yy, exp_y);
445 j++;
446 }
447 }
448 }
449
450 return P;
451}

454gsMatrix<> get_coefficient(const gsTHBSplineBasis<2>& basis,
455 const gsMatrix<>& parameters,
456 const gsMatrix<>& points,
457 const int k,
458 int degreeLS,
459 const std::vector<index_t>& indicesLS,
460 const real_t sigma)
461{
462 typedef gsEigen::Matrix<real_t, Dynamic, Dynamic> matrix_t;
463
464 while (true)
465 {
466 gsMatrix<> A;
467 gsMatrix<> B;
468
469 assemble_regular_LS_system(basis, k, indicesLS, parameters, points,
470 degreeLS, A, B);
471
472
473 gsEigen::JacobiSVD<matrix_t> svdA(A.transpose(),
474 gsEigen::ComputeThinU | gsEigen::ComputeThinV);
475 gsMatrix<> coefficientsLS = svdA.solve(B);
476
477 // std::cout << k << " -> " << std::endl;
478
479 // ------------------------------------------------------------
480 gsMatrix<> gridPoints;
481 int row_index = -1;
482
483
484 gsMatrix<> S = assemble_system_coefsLS_to_basis_fun(basis, k, gridPoints,
485 row_index);
486 const gsMatrix<> support = basis.support(k);
487 gsMatrix<> P = polynomial_basis_evaluation(degreeLS, gridPoints, support);
488
489 gsMatrix<> rhs = P * coefficientsLS;
490
491 gsMatrix<> coefficients = S.fullPivHouseholderQr().solve(rhs);
492 gsMatrix<> coefficient = coefficients.row(row_index);
493
494 // gsInfo << std::setprecision(16) << coefficient << "\n";
495
496 // ------------------------------------------------------------
497
498 //std::cout << "size S: " << S.rows() << " x " << S.cols() << "\n"
499 // << "size P: " << P.rows() << " x " << P.cols() << "\n"
500 // << "size A: " << A.rows() << " x " << A.cols() << "\n";
501 gsMatrix<> AAT_1 = (A * A.transpose()).inverse();
502 gsMatrix<> G = S.inverse() * P * AAT_1 * A;
503
504 // std::cout << "size G: " << G.rows() << " x " << G.cols() << "\n";
505 gsEigen::JacobiSVD<matrix_t> svdG(G);
506 real_t largest_singular_value = svdG.singularValues()(0);
507
508 if (sigma < largest_singular_value)
509 {
510 degreeLS--;
511 if (degreeLS == -1)
512 {
513 GISMO_ERROR("Choose a larger value of sigma!");
514 return coefficient;
515 }
516 }
517 else
518 {
519 //std::cout << "degree: " << degreeLS << std::endl;
520 return coefficient;
521 }
522 }
523}
