18template <typename E>
19inline void quadrature(const gismo::expr::_expr<E> & ee,
20 gsMatrix<T> & lm)
21{
22 auto & a = ee.rowVar();
23 auto & b = ee.colVar();
24 const index_t ca = a.cardinality();
25 const index_t cb = b.cardinality();
26
27 if (E::isMatrix())
28 lm.setZero(ca,cb);
29 else if (E::isVector())
30 lm.setZero(ca,1);
31 else if (E::isRowVector())
32 lm.setZero(1,cb);
33 else
34 GISMO_ERROR("Something went terribly wrong at this point.");
35
36 // ------- Assembly Core -------
37 const T * w;
38 for (index_t i=0; i!=a.cardinality(); ++i)
39 for (index_t j=0; j!=b.cardinality(); ++j)
40 {
41 w = m_quWeights.data();
42 for (index_t k = 0; k != m_quWeights.rows(); ++k)
43 lm(i,j) += (*(++w)) * ee.eval(i,j,k).value();
44 }
45}
