30template<class T>
31void gsWriteParaviewMultiPhysics(std::map<std::string, const gsField<T>*> fields,
32 std::string const & fn,
33 unsigned npts, bool mesh, bool ctrlNet)
34{
35 gsDebugVar(fn);
36 const unsigned numP = fields.begin()->second->patches().nPatches();
37 gsParaviewCollection collection(fn);
38 std::string baseName = gsFileManager::getFilename(fn); // file name without a path
39
40 for ( unsigned i=0; i < numP; ++i )
41 {
42 const gsBasis<> & dom = fields.begin()->second->isParametrized() ?
43 fields.begin()->second->igaFunction(i).basis() : fields.begin()->second->patch(i).basis();
44
45 gsWriteParaviewMultiPhysicsSinglePatch( fields, i, fn + util::to_string(i), npts);
46 collection.addPart(baseName + util::to_string(i) + ".vts", -1, "Solution", i );
47
48 if ( mesh )
49 {
50 writeSingleCompMesh(dom, fields.begin()->second->patch(i), fn + util::to_string(i) + "_mesh");
51 collection.addPart(baseName + util::to_string(i) + "_mesh" + ".vtp",-1, "Mesh", i);
52 }
53 if ( ctrlNet ) // Output the control net
54 {
55 writeSingleControlNet(fields.begin()->second->patch(i), fn + util::to_string(i) + "_cnet");
56 collection.addPart(baseName + util::to_string(i) + "_cnet" + ".vtp", -1, "Mesh", i);
57 }
58
59 }
60 collection.save();
61}

77template<class T>
78void gsWriteParaviewMultiPhysicsSinglePatch(std::map<std::string,const gsField<T> *> fields,
79 const unsigned patchNum,
80 std::string const & fn,
81 unsigned npts)
82{
83 const gsGeometry<> & geometry = fields.begin()->second->patches().patch(patchNum);
84 const short_t n = geometry.targetDim();
85 const short_t d = geometry.domainDim();
86
87 gsMatrix<> ab = geometry.support();
88 gsVector<> a = ab.col(0);
89 gsVector<> b = ab.col(1);
90 gsVector<unsigned> np = distributePoints<T>(geometry,npts);
91 gsMatrix<> pts = gsPointGrid(a,b,np);
92
93 gsMatrix<> eval_geo = geometry.eval(pts);
94 std::map<std::string, gsMatrix<> > data;
95 for (typename std::map<std::string,const gsField<T> *>::iterator it = fields.begin(); it != fields.end(); it++)
96 {
97 data[it->first] = it->second->isParametric() ?
98 it->second->function(patchNum).eval(pts) : it->second->function(patchNum).eval(eval_geo);
99
100 if ( data[it->first].rows() == 2 )
101 {
102 data[it->first].conservativeResize(3,eval_geo.cols() );
103 data[it->first].row(2).setZero();
104 }
105 }
106
107 if (3 -d > 0)
108 {
109 np.conservativeResize(3);
110 np.bottomRows(3-d).setOnes();
111 }
112 else if (d > 3)
113 {
114 gsWarn<< "Cannot plot 4D data.\n";
115 return;
116 }
117
118 if ( 3 - n > 0 )
119 {
120 eval_geo.conservativeResize(3,eval_geo.cols() );
121 eval_geo.bottomRows(3-n).setZero();
122 }
123 else if (n > 3)
124 {
125 gsWarn<< "Data is more than 3 dimensions.\n";
126 }
127
128 /*for (typename std::map<std::string, gsMatrix<> >::iterator it = data.begin(); it != data.end(); it++)
129 {
130 if ( it->second.rows() > 1 )
131 {
132 it->second.conservativeResize(3,eval_geo.cols() );
133 it->second.bottomRows( 3-dd ).setZero();
134 }
135 }*/
136
137 gsWriteParaviewMultiTPgrid(eval_geo, data, np.template cast<index_t>(), fn);
138}

140template<class T>
141void gsWriteParaviewMultiTPgrid(gsMatrix<T> const& points,
142 std::map<std::string, gsMatrix<T> >& data,
143 const gsVector<index_t> & np,
144 std::string const & fn)
145{
146 const int n = points.rows();
147
148 std::string mfn(fn);
149 mfn.append(".vts");
150 std::ofstream file(mfn.c_str());
151 file << std::fixed; // no exponents
152 file << std::setprecision (PLOT_PRECISION);
153
154 file <<"<?xml version=\"1.0\"?>\n";
155 file <<"<VTKFile type=\"StructuredGrid\" version=\"0.1\">\n";
156 file <<"<StructuredGrid WholeExtent=\"0 "<< np(0)-1<<" 0 "<<np(1)-1<<" 0 "
157 << (np.size()>2 ? np(2)-1 : 0) <<"\">\n";
158 file <<"<Piece Extent=\"0 "<< np(0)-1<<" 0 "<<np(1)-1<<" 0 "
159 << (np.size()>2 ? np(2)-1 : 0) <<"\">\n";
160
161 file <<"<PointData>\n";
162 for (typename std::map<std::string, gsMatrix<T> >::iterator it = data.begin(); it != data.end(); it++)
163 {
164 file <<"<DataArray type=\"Float32\" Name=\""<< it->first <<"\" format=\"ascii\" NumberOfComponents=\""<< ( it->second.rows()==1 ? 1 : 3) <<"\">\n";
165 if ( it->second.rows()==1 )
166 for ( index_t j=0; j<it->second.cols(); ++j)
167 file<< it->second.at(j) <<" ";
168 else
169 {
170 for ( index_t j=0; j<it->second.cols(); ++j)
171 {
172 for ( index_t i=0; i!=it->second.rows(); ++i)
173 file<< it->second(i,j) <<" ";
174 for ( index_t i=it->second.rows(); i<3; ++i)
175 file<<"0 ";
176 }
177 }
178 file <<"</DataArray>\n";
179 }
180 file <<"</PointData>\n";
181 file <<"<Points>\n";
182 file <<"<DataArray type=\"Float32\" NumberOfComponents=\"3\">\n";
183 for ( index_t j=0; j<points.cols(); ++j)
184 {
185 for ( index_t i=0; i!=n; ++i)
186 file<< points(i,j) <<" ";
187 for ( index_t i=n; i<3; ++i)
188 file<<"0 ";
189 }
190 file <<"</DataArray>\n";
191 file <<"</Points>\n";
192 file <<"</Piece>\n";
193 file <<"</StructuredGrid>\n";
194 file <<"</VTKFile>\n";
195
196 file.close();
197}
