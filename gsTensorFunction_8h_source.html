137 void eval_into(const gsMatrix<T>& u, gsMatrix<T>& result) const
138 {
139 const index_t n = m_ranks.size();
140 const index_t d = u.rows(); //Dimension of source space
141 result.resize(n, u.cols() );
142
143 std::vector<gsMatrix<T> > ev(d);
144 for (index_t j=0; j!= u.cols(); ++j) //eval points
145 {
146 for (index_t k=0; k!= d; ++k) //Dimension of source space/Number of univariate functions
147 m_skeleton[k]->eval_into(u.col(j).row(k), ev[k]);
148
149 // Hadamard product
150 gsMatrix<T> & Hprod = ev.front();
151 for (index_t i=1; i!=d; ++i)
152 Hprod.array() *= ev[i].array();
153
154 index_t pos = 0;
155 for (index_t k=0; k!=n; ++k)
156 {
157 result(k,j) = Hprod.middleRows(pos, m_ranks[k]).sum();
158 pos += m_ranks[k];
159 }
160 }
161 }
