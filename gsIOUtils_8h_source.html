39  std::ofstream& out)
40 {
41  // The format is (similar for d < 3)
42  //
43  // - dimension of geometry space, whether or not the volume is
44  // rational: 1=rational, 0=non-rational
45  // - the number of coefficients in the first parameter direction,
46  // the polynomial order in this direction (i.e. degree+1)
47  // - the knot vector in the first parameter direction, multiple
48  // knots are represented by giving the knot value several times
49  // - the number of coefficients in the second parameter direction,
50  // the polynomial order in this direction (i.e. degree+1)
51  // - the knot vector in the second parameter direction
52  // - the number of coefficients in the third parameter direction,
53  // the polynomial order in this direction (i.e. degree+1)
54  // - the knot vector in the third parameter direction
55  // - the volume coefficients
56 
57 
58 
59  out << std::setprecision(15);
60 
61  out << bspl.geoDim() << " " << 0 << "\n";
62 
63  for (short_t dim = 0; dim < bspl.parDim(); dim++)
64  {
65  out << bspl.basis().size(dim) << " " <<
66  bspl.basis().degree(dim) + 1 << "\n";
67 
68  const gsKnotVector<T> kv = bspl.basis().knots(dim);
69  typename gsKnotVector<T>::const_iterator iter;
70  for (iter = kv.begin(); iter != kv.end(); iter++)
71  {
72  out << *iter << " ";
73  }
74  out << "\n";
75  }
76 
77  const gsMatrix<T>& coefs = bspl.coefs();
78 
79  for (int row = 0; row < coefs.rows(); row++)
80  {

82 
83  }

260 void makeMesh(const gsBasis<T>& basis, gsMesh<T> & mesh, int n = 0)
261 {
262  const unsigned d = basis.dim();
263 
264  typedef typename gsMesh<T>::VertexHandle Vertex;
265  typename gsBasis<T>::domainIter domIter = basis.makeDomainIterator();
266 
267  // variables for iterating over a cube (element is a cube)
268  const gsVector<unsigned> zeros = gsVector<unsigned>::Zero(d);
269  const gsVector<unsigned> ones = gsVector<unsigned>::Ones(d);
270  gsVector<unsigned> cur;
271 
272  // maps integer representation of a vertex into pointer to the
273  // vertex coordinates
274  std::vector<Vertex> map(1ULL<<d);
275 
276  // neighbour[i] are integer representations of certain neighbours of
277  // vertex i (i counts in lexicographics order over all vertices)
278  std::vector<std::vector<unsigned> > neighbour(1ULL<<d,
279  std::vector<unsigned>() );
280 
281  cur.setZero(d);
282  int counter = 0;
283  do
284  {
285  // set neighbour
286  for (unsigned dim = 0; dim < d; dim++)
287  {
288  if (cur(dim) == 0)
289  {
290  const unsigned tmp = counter | (1<< dim) ;
291  neighbour[counter].push_back(tmp);
292  }
293  }
294  counter++;
295 
296  } while (nextCubePoint<gsVector<unsigned> >(cur, zeros, ones));
297 
298  gsVector<T> vertex(d);
299 
300  for (; domIter->good(); domIter->next())
301  {
302  const gsVector<T>& low = domIter->lowerCorner();
303  const gsVector<T>& upp = domIter->upperCorner();
304  const T vol = domIter->volume();
305 
306  vertex.setZero();
307  cur.setZero();
308  counter = 0;
309 
310  // add points to the mesh
311  do
312  {
313  // get appropriate coordinate of a point
314  for (unsigned dim = 0; dim < d; dim++)
315  {
316  vertex(dim) = ( cur(dim) ? upp(dim) : low(dim) );
317  }
318 
319  Vertex v = mesh.addVertex(vertex);
320  v->data = vol;
321  map[counter++] = v;
322 
323  } while (nextCubePoint<gsVector<unsigned> >(cur, zeros, ones));
324 
325 
326  // add edges to the mesh (connect points)
327  for (size_t index = 0; index != neighbour.size(); index++)
328  {
329  const std::vector<unsigned> & v = neighbour[index];
330 
331  for (size_t ngh = 0; ngh != v.size(); ngh++)
332  {
333  // add more vertices (n) for better physical resolution
334  mesh.addLine( map[index], map[v[ngh]], n );
335  //mesh.addEdge( map[index], map[v[ngh]] );
336  }
337  }
338 
339  // idea: instead of edges add the faces to the mesh
340  // mesh->addFace( mesh.vertices().back(),
341  // *(mesh.vertices().end()-3),
342  // *(mesh.vertices.end()-4),
