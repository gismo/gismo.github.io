56 void eval_matrix_into(const gsMatrix<T>& u, gsMatrix<T>& result, index_t k = 0) const
57 {
58 const index_t sz = (u.rows()+1)*u.rows()/2;
59 result.resize(sz*sz,u.cols());
60
61
62 gsMatrix<T> angles, Phi, G;
63 angles = m_alpha->piece(k).eval(u);
64 G = m_G->piece(k).eval(u);
65 // composite linear elasticity tensor
66 for (index_t j=0; j!=u.cols(); j++)
67 {
68 _makePhi(angles.col(k),Phi);
69 result.reshapeCol(j,sz,sz) = Phi.transpose() * G.reshapeCol(j,sz,sz) * Phi;
70 }

89 void _makePhi( const gsMatrix<T> angles,
90 gsMatrix<T> & Phi) const
91 {
92 if (angles.rows()==1) // 2d elasticity
93 {
94 Phi.resize(3,3);
95 Phi.setZero();
96
97 const T & alpha = angles(0,0);
98
99 // Make transformation matrix
100 Phi(0,0) = Phi(1,1) = math::pow(math::cos(alpha),2);
101 Phi(0,1) = Phi(1,0) = math::pow(math::sin(alpha),2);
102 Phi(2,0) = Phi(0,2) = Phi(2,1) = Phi(1,2) = math::sin(alpha) * math::cos(alpha);
103 Phi(2,0) *= -2.0;
104 Phi(2,1) *= 2.0;
105 Phi(1,2) *= -1.0;
106 Phi(2,2) = math::pow(math::cos(alpha),2) - math::pow(math::sin(alpha),2);
107 }
108 else if (angles.rows()==3)
109 {
110 Phi.resize(6,6);
111 Phi.setZero();
112
113 const T & alpha = angles(0,0);
114 const T & beta = angles(1,0);
115 const T & gamma = angles(2,0);
116
117 gsMatrix<T,3,3> Ax,Ay,Az,A;
118 Ax << 1, 0, 0,
119 0, math::cos(alpha), -math::sin(alpha),
120 0, math::sin(alpha), math::cos(alpha);
121
122 Ay << math::cos(beta), 0, math::sin(beta),
123 0, 1, 0,
124 -math::sin(beta), 0, math::cos(beta);
125
126 Az << math::cos(gamma), -math::sin(gamma), 0,
127 math::sin(gamma), math::cos(gamma), 0,
128 0, 0, 1;
129
130 // Tiat-Bryan angle (X1,Y2,Z3) --> https://en.wikipedia.org/wiki/Euler_angles
131 Ax *= Ay;
132 Ax *= Az;
133 A.swap(Ax);
134
135 /*
136 Phi:
137 a11^2 ,a12^2 ,a13^2 ,2*a12*a11 ,2*a12*a13 ,2*a11*a13
138 a21^2 ,a22^2 ,a23^2 ,2*a21*a22 ,2*a22*a23 ,2*a21*a23
139 a31^2 ,a32^2 ,a33^2 ,2*a31*a32 ,2*a32*a33 ,2*a31*a33
140 a11*a21 ,a12*a22 ,a13*a23 ,a11*a22+a12*a21 ,a12*a23+a13*a22 ,a11*a23+a13*a21
141 a21*a31 ,a22*a32 ,a23*a33 ,a21*a32+a22*a31 ,a22*a33+a23*a32 ,a21*a33+a23*a31
142 a11*a31 ,a12*a32 ,a13*a33 ,a11*a32+a31*a12 ,a12*a33+a13*a32 ,a11*a33+a31*a13
143 */
144
145 Phi(0,0) = A(0,0)*A(0,0);
146 Phi(0,1) = A(0,1)*A(0,1);
147 Phi(0,2) = A(0,2)*A(0,2);
