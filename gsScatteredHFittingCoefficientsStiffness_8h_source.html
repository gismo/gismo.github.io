100std::set<index_t> check_rectangle(const gsTensorBSplineBasis<2>& tensBasis,
101 std::set<index_t>& indices)
102{
103 gsVector<index_t, 2> tmpIdx;
104 std::set<index_t> indices_u;
105 std::set<index_t> indices_v;
106 std::set<index_t> check_u;
107 std::set<index_t> check_v;
108 std::set<index_t> check;
109
110 for(std::set<index_t>::iterator it=indices.begin(); it!=indices.end(); ++it)
111 {
112 tmpIdx = tensBasis.tensorIndex(*it);
113 indices_u.insert(tmpIdx(0));
114 indices_v.insert(tmpIdx(1));
115 }
116
117 for(index_t u = *indices_u.begin(); u <= *(--indices_u.end()); ++u)
118 check_u.insert(u);
119 for(index_t v = *indices_v.begin(); v <= *(--indices_v.end()); ++v)
120 check_v.insert(v);
121
122 for(std::set<index_t>::iterator it = check_u.begin(); it != check_u.end(); ++it)
123 for(std::set<index_t>::iterator jt = check_v.begin(); jt != check_v.end(); ++jt)
124 check.insert(tensBasis.index(*it,*jt));
125
126 return check;
127}

220std::set<index_t> get_data_inside_knot_span(const gsMatrix<>& parameters,
221 const gsMatrix<>& support,
222 const gsMatrix<>& center,
223 const int k,
224 const gsKnotVector<>& kv0,
225 const gsKnotVector<>& kv1,
226 real_t& u_diff,
227 real_t& v_diff)
228{
229 std::set<index_t> indices;
230
231 if( k == 1 )
232 {
233 u_diff = std::fabs(support(0,0) - support(0,1)) / 2;
234 v_diff = std::fabs(support(1,0) - support(1,1)) / 2;
235
236 indices = get_data_inside_rectangle(parameters, center, u_diff, v_diff);
237 }
238
239 else
240 {
241 // Assuming uniform knot vectors:
242 u_diff += kv0.maxIntervalLength();
243 v_diff += kv1.maxIntervalLength();
244
245 if(u_diff == 0 || v_diff == 0)
246 {
247 gsInfo << "Warning: error due to rounding";
248 }
249
250 indices = get_data_inside_rectangle(parameters, center, u_diff, v_diff);
251 }
252
253 return indices;
254}

258void knots_within_iterval(gsKnotVector<real_t>& knots,
259 const real_t left,
260 const real_t right)
261{
262 if( left == right )
263 gsInfo << "Warning: No existing interval: " << "[" << left << ", " << right << "]" << std::endl;
264
265 std::vector<real_t> repeated_knots;
266 for(typename gsKnotVector<real_t>::const_iterator it=knots.begin(); it!=knots.end(); ++it)
267 if( (*it > left || almostEqual(*it,left) ) &&
268 (*it < right || almostEqual(*it,right) ) )
269 repeated_knots.push_back(*it);
270
271 knots = gsKnotVector<real_t>(repeated_knots);
272}

294int parameters_inside_pieces(const gsMatrix<>& parameters,
295 const std::set<index_t>& indices,
296 gsKnotVector<real_t>& uKnots,
297 gsKnotVector<real_t>& vKnots)
298{
299 int pieces = 0;
300
301 if( uKnots.size() == 0 || vKnots.size() == 0)
302 {
303 std::cerr << "Warning: error due to rounding." << std::endl;
304 return pieces;
305 }
306
307 for(size_t u = 0; u < uKnots.size(); u++)
308 {
309 for(size_t v = 0; v < vKnots.size(); v++)
310 {
311 for(std::set<index_t>::const_iterator it = indices.begin(); it != indices.end(); ++it)
312 {
313 real_t x = parameters(0,*it);
314 real_t y = parameters(1,*it);
315
316 if((uKnots[u]<=x)&&(x<=uKnots[u+1])&&(vKnots[v]<=y)&&(y<=vKnots[v+1]))
317 {
318 pieces++;
319 break;
320 }
321 }
322 }
323 }
324 return pieces;
325}

332real_t parameters_distribution(const gsTensorBSplineBasis<2, real_t>& tensBasis,
333 const gsMatrix<>& parameters,
334 const std::set<index_t>& indices,
335 const gsMatrix<>& center,
336 const real_t half_b,
337 const real_t half_h,
338 const int basis_fun_index)
339{

341 knots_perimeter(tensBasis, center, half_b, half_h, uPieces, vPieces);
342

345
346 return numPieces/totPieces;

352std::set<index_t> indices_for_fitting(const gsTHBSplineBasis<2, real_t>& basis,
353 const int basis_fun_index,
354 const gsMatrix<>& parameters,
355 const int cp_magnitude,
356 const real_t hole,
357 const int circle,
358 const unsigned degree,
359 unsigned fitting_deg,
360 std::vector<index_t>& enlargement)
361{
362 std::set<index_t> indices;
363
364 const gsMatrix<> support = basis.support(basis_fun_index);
365 const gsMatrix<> center = get_center(support);
366 real_t radius = (support.col(0) - support.col(1)).norm() / 2;
367 real_t u_diff = std::fabs(support(0,0) - support(0,1)) / 2;
368 real_t v_diff = std::fabs(support(1,0) - support(1,1)) / 2;
369 const int level = basis.levelOf(basis_fun_index);
370 const gsKnotVector<>& kv0 = basis.getBases()[level]->knots(0);
371 const gsKnotVector<>& kv1 = basis.getBases()[level]->knots(1);
372 const gsTensorBSplineBasis<2>& tbasis = *basis.getBases()[level];
373
374 int k = 1;
375 int num_points = 0;
376 real_t distribPar = 0;
377 while( ( num_points <= cp_magnitude ) || ( distribPar < hole ) )
378 {
379 if(circle == 0)
380 {
381 indices = get_data_inside_circle(parameters, center, radius * k);
382 distribPar = 1;
383 }
384 else if(circle == 1)
385 {
386 indices = get_data_inside_rectangle(parameters, center, u_diff * k, v_diff * k );
387 if( distribPar < hole )
388 {
389 if( u_diff == 0 || v_diff == 0 )
390 {
391 gsInfo << "Warning: error due to rounding" << std::endl;
392 break;
393 }
394 else
395 distribPar = parameters_distribution(tbasis, parameters, indices, center, u_diff * k, v_diff * k, basis_fun_index);
396 }
397 }
398 else
399 {
400 indices = get_data_inside_knot_span(parameters, support, center, k, kv0, kv1, u_diff, v_diff);
401 if( distribPar < hole )
402 {
403 distribPar = parameters_distribution(tbasis, parameters, indices, center, u_diff, v_diff, basis_fun_index);
404 }
405 }
406
407 num_points = indices.size();
408 k++;
409 }
410
411 enlargement.push_back(num_points);
412
413 int d = degree;
414 while( ( num_points < (d + 1)*(d + 2)/2 )&&( 0 < d ) )
415 d -= 1;
416 fitting_deg = d;
417
418 return indices;
419}
