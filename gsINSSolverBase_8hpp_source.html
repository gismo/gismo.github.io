47template<class T>
48void gsINSSolverBase<T>::applySolver(gsMatrix<T>& solution, real_t alpha_u, real_t alpha_p)
49{
50 GISMO_ASSERT(solution.rows() > 0,"The solution in applySolver() is empty!");
51
52 int usize = getAssembler()->getPshift();
53 int pdofs = getAssembler()->getPdofs();
54
55 m_clock.restart();
56 m_solver.factorize(getAssembler()->matrix());
57 m_solsetupT += m_clock.stop();
58
59 m_clock.restart();
60 gsMatrix<T> newsol = m_solver.solve(getAssembler()->rhs());
61 solution.topRows(usize) = alpha_u * newsol.topRows(usize) + (1 - alpha_u)*solution.topRows(usize);
62 solution.bottomRows(pdofs) = alpha_p * newsol.bottomRows(pdofs) + (1 - alpha_p)*solution.bottomRows(pdofs);
63 m_solveT += m_clock.stop();
64}

67template<class T>
68void gsINSSolverBase<T>::solve(const int maxIterations, const T epsilon, const int minIterations)
69{
70 GISMO_ASSERT(getAssembler()->isInitialized(), "Assembler must be initialized first, call initialize()");
71 int iter = 0;
72 m_relNorm = solutionChangeRelNorm();
73
74 while ((iter < minIterations) || ((m_relNorm > epsilon) && (iter < maxIterations)))
75 {
76 gsInfo << "Iteration number " << m_iterationNumber + 1 << "...";
77
78 nextIteration();
79 m_relNorm = solutionChangeRelNorm();
80 gsInfo << " Solution change relative norm: " << m_relNorm << "\n";
81
82 iter++;
83 }
84}

108 T relNorm;

112 gsMatrix<T> solChangeVector = getAssembler()->getSolution() - m_solution;
113 relNorm = solChangeVector.norm() / m_solution.norm();

230void gsINSSolverBaseIter<T,SolverType>::applySolver(gsMatrix<T>& solution)
231{
232 m_matrices.at("matNS") = m_assemblerRef.matrix();
233 gsMatrix<T> rhs = m_assemblerRef.rhs();
234
235 updatePrecMat();
236
237 if (m_precType.substr(0, 2) == "AL")
238 {
239 m_matrices.at("matNS") += m_matGammaPart;
240 rhs += m_rhsGammaPart;
241 }
242
243 m_clock.restart();
244 m_pPrec->update(m_matrices);
245
246 SolverType solver(m_matrices.at("matNS"), m_pPrec);

249 m_solsetupT += m_clock.stop();
250
251 //solution.setZero(); // zero initial solution for iterative solver in each iteration
252
253 m_clock.restart();
254 solver.solve(rhs, solution);

256
257 m_linIterVector.push_back(solver.iterations());
