39inline std::ostream& formatTime(std::ostream& os, double sec)
40{
41 int flo = (int)(sec);
42 sec -= flo;
43 int hh = flo / 3600;
44 flo = flo % 3600;
45 int mm = flo / 60;
46 double ss = (flo % 60) + sec;
47 if (hh > 0) os << hh << "h ";
48 if (mm > 0) os << mm << "m ";
49 std::streamsize prec = os.precision();
50 os.precision(2);
51 os << ss << " s";
52 os.precision(prec);
53 return os;
54}
