39template<class T>
40void gsINSAssemblerUnsteady<T>::fillBase()
41{
42 const T invTimeStep = 1. / m_timeStepSize;
43 int numDofs = Base::numDofs();
44 int uDofs = Base::getUdofs();
45 int tarDim = Base::getTarDim();
46
47 gsSparseMatrix<T> stokesMatrix(numDofs, numDofs);
48
49 m_blockAssembler.fillStokesSystem_into(stokesMatrix, m_baseRhs);
50
51 gsVector<int> nonZerosPerColumnVector;
52 nonZerosPerColumnVector.setZero(numDofs);
53 for (int s = 0; s < tarDim; ++s)
54 for (int i = 0; i < uDofs; i++)
55 nonZerosPerColumnVector(i + s * uDofs) = m_blockAssembler.getBlockNpattern().col(i).nonZeros();
56
57 gsSparseMatrix<T> blockNpatternMatrix(numDofs, numDofs);
58 blockNpatternMatrix.reserve(nonZerosPerColumnVector);
59
60 #pragma omp parallel for num_threads(m_blockAssembler.getNumThreads())
61 for (index_t col = 0; col < uDofs; ++col)
62 for (typename gsSparseMatrix<T>::InnerIterator it(m_blockAssembler.getBlockNpattern(), col); it; ++it)
63 for (index_t s = 0; s < tarDim; ++s)
64 blockNpatternMatrix.insert(it.row() + s * uDofs, it.col() + s * uDofs) = 0.;
65
66 m_baseMatrix = stokesMatrix + blockNpatternMatrix;
67
68 if (m_blockAssembler.isUnsteady())
69 {
70 #pragma omp parallel for num_threads(m_blockAssembler.getNumThreads())
71 for (index_t col = 0; col < uDofs; ++col)
72 for (typename gsSparseMatrix<T>::InnerIterator it(m_blockAssembler.getBlockM(), col); it; ++it)
73 for (index_t s = 0; s < tarDim; ++s)
74 m_baseMatrix.coeffRef(it.row() + s * uDofs, it.col() + s * uDofs) += invTimeStep * it.value();
75 }
76
77 m_bMatrixReady = false;
78 m_bRhsReady = false;
79}

82template<class T>
83void gsINSAssemblerUnsteady<T>::fillMatrix()
84{
85 int uDofs = Base::getUdofs();
86
87 m_matrix = m_baseMatrix;
88 #pragma omp parallel for num_threads(m_blockAssembler.getNumThreads())
89 for (index_t col = 0; col < uDofs; ++col)
90 for (typename gsSparseMatrix<T>::InnerIterator it(m_blockAssembler.getBlockN(), col); it; ++it)
91 for (index_t s = 0; s < Base::getTarDim(); ++s)
92 m_matrix.coeffRef(it.row() + s*uDofs, it.col() + s*uDofs) += it.value();
93
94 m_bMatrixReady = true;
95
96 if (!m_matrix.isCompressed())
97 m_matrix.makeCompressed();
98}

101template<class T>
102void gsINSAssemblerUnsteady<T>::fillRhs()
103{
104 const T invTimeStep = 1. / m_timeStepSize;
105 int uDofs = Base::getUdofs();
106
107 m_rhs.noalias() = m_baseRhs + m_blockAssembler.getRhsN();
108
109 if (m_blockAssembler.isUnsteady())
110 {
111 #pragma omp parallel for num_threads(m_blockAssembler.getNumThreads())
112 for (index_t s = 0; s < Base::getTarDim(); ++s)
113 m_rhs.middleRows(s * uDofs, uDofs).noalias() += invTimeStep * m_blockAssembler.getBlockM() * m_blockAssembler.getSolution().middleRows(s * uDofs, uDofs);
114 }
115
116 m_bRhsReady = true;
117}
