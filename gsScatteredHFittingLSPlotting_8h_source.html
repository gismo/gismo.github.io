27void generate_points_whole_domain(const gsTHBSplineBasis<2>& basis,
28 const int basis_fun_index,
29 gsMatrix<>& points,
30 const int level
31 ) // For fitting LS surface
32{
33 //const int level = basis.levelOf(basis_fun_index);
34 const gsTensorBSplineBasis<2>& ten_basis = *basis.getBases()[level];
35 const int degree = ten_basis.maxDegree();
36
37 const gsMatrix<> support = basis.support(basis_fun_index);
38
39 const real_t factor = 1.0 / (2 * (degree + 1));
40 const gsMatrix<> diff = factor * (support.col(1) - support.col(0));
41
42 //const int n = 14; to fix in accordance to the considered function.
43 //gsInfo << "Basis max degree: " << degree << std::endl;
44 const int n = ten_basis.component(0).size();
45
46 gsVector<unsigned> num_points(2);
47 num_points << n, n;
48
49 gsVector<> start(2);
50 start << 0, 0;
51 start(0) += diff(0, 0);
52 start(1) += diff(1, 0);
53
54 gsVector<> end(2);
55 end << 1, 1;
56 end(0) -= diff(0, 0);
57 end(1) -= diff(1, 0);
58
59 points = gsPointGrid(start, end, num_points);
60}

68gsMatrix<> assemble_system_coefsLS_to_tensorbasis_fun(const gsTHBSplineBasis<2>& basis,
69 const int basis_fun_index,
70 gsMatrix<>& gridPoints,
71 gsMatrix<index_t>& indices,
72 int& basis_fun_tensor_index,
73 std::set<index_t>& employed_functions,
74 const int level,
75 bool whole_domain
76 )
77{
78 //const int level = basis.levelOf(basis_fun_index);
79 const gsTensorBSplineBasis<2>& ten_basis = *basis.getBases()[level];
80 const unsigned tensor_index = basis.flatTensorIndexOf(basis_fun_index);
81
82 gsMatrix<> tmpPoints;
83
84 if(whole_domain)
85 generate_points_whole_domain(basis, basis_fun_index, tmpPoints, level);
86 else
87 tmpPoints = generate_points(basis, basis_fun_index);
88
89 ten_basis.active_into(tmpPoints, indices);
90
91 std::set<index_t> set_indices = matrix_to_set(indices);
92 std::map<index_t, index_t> basis_to_cols = compute_ordering(set_indices);
93
94 gsMatrix<real_t> domain(2,2);
95 domain << 0, 1, 0, 1;
96
97 if(whole_domain)
98 gridPoints = computeGridPoints(ten_basis, domain, set_indices);
99 else
100 gridPoints = computeGridPoints(ten_basis, basis.support(basis_fun_index),
101 set_indices);
102
103 gsMatrix<> eval;
104 ten_basis.eval_into(gridPoints, eval);
105 ten_basis.active_into(gridPoints, indices);
106
107
108 gsMatrix<> A(gridPoints.cols(), set_indices.size());
109 A.setZero();
110
111
112 for (int point_index = 0; point_index != eval.cols(); point_index++)
113 {
114 for (int index = 0; index != eval.rows(); index++)
115 {
116 const real_t basis_value = eval(index, point_index);
117 const unsigned basis_index = indices(index, point_index);
118 if (basis_to_cols.find(basis_index) != basis_to_cols.end())
119 {
120 const int col = basis_to_cols[basis_index];
121 A(point_index, col) = basis_value;
122 employed_functions.insert(basis_index);
123 }
124 }
125 }
126
127
128
129 basis_fun_tensor_index = basis_to_cols[tensor_index];
130 return A;
131}

137gsMatrix<> get_coefficient_pol(const gsTHBSplineBasis<2>& basis,
138 const gsMatrix<>& parameters,
139 const gsMatrix<>& points,
140 const int k,
141 int degreeLS,
142 const std::vector<index_t>& indicesLS,
143 const real_t sigma,
144 int iteration,
145 const index_t cp_num,
146 const int it_max
147 )
148{
149 typedef gsEigen::Matrix<real_t, Dynamic, Dynamic> matrix_t;
150
151
152
153 while (true)
154 {
155 gsMatrix<> A;
156 gsMatrix<> B;
157
158 assemble_regular_LS_system(basis, k, indicesLS, parameters, points,
159 degreeLS, A, B);
160
161
162 gsEigen::JacobiSVD<matrix_t> svdA(A.transpose(),
163 gsEigen::ComputeThinU | gsEigen::ComputeThinV);
164 gsMatrix<> coefficientsLS = svdA.solve(B);
165
166 // std::cout << k << " -> " << std::endl;
167
168 // ------------------------------------------------------------
169 gsMatrix<> gridPoints;
170 gsMatrix<> gridPoints_ls;
171 gsMatrix<index_t> indices;
172 gsMatrix<index_t> indices_ls;
173 std::set<index_t> employed_functions;
174 std::set<index_t> employed_functions_ls;
175 int row_index = -1;
176 int row_index_ls = -1;
177
178 const int level=basis.levelOf(k);
179
180 gsMatrix<> S = assemble_system_coefsLS_to_tensorbasis_fun(basis, k, gridPoints, indices, row_index, employed_functions, level, false);
181 gsMatrix<> W = assemble_system_coefsLS_to_tensorbasis_fun(basis, k, gridPoints_ls, indices_ls, row_index_ls, employed_functions_ls, 0, true);
182
183 const gsMatrix<> support = basis.support(k);
184 gsMatrix<> P = polynomial_basis_evaluation(degreeLS, gridPoints, support);
185
186 gsMatrix<> support_w(2,2);
187 support_w << 0, 1, 0, 1;
188 gsMatrix<> P_ls = polynomial_basis_evaluation(degreeLS, gridPoints_ls, support);
189
190 gsMatrix<> rhs = P * coefficientsLS;
191 gsMatrix<> rhs_ls = P_ls * coefficientsLS;
192
193
194 // S*coefficients = P*coefficientsLS
195 // W*coefficients_ls = P_ls*coefficientsLS
196
197 gsMatrix<> coefficients = S.fullPivHouseholderQr().solve(rhs);
198 gsMatrix<> coefficients_ls = W.fullPivHouseholderQr().solve(rhs_ls);
199
200
201
202 gsMatrix<> coefficient = coefficients.row(row_index);
203
204
205 gsMatrix<> AAT_1 = (A * A.transpose()).inverse();
206 gsMatrix<> G = S.inverse() * P * AAT_1 * A;
207
208
209 gsEigen::JacobiSVD<matrix_t> svdG(G);
210 real_t largest_singular_value = svdG.singularValues()(0);
211
212 if (sigma < largest_singular_value)
213 {
214 degreeLS--;
215 if (degreeLS == -1)
216 {
217 GISMO_ERROR("Choose a larger value of sigma!");
218 return coefficient;
219 }
220 }
221 else
222 {
223 if( k == cp_num && iteration == it_max - 1 )
224 {
225 gsMatrix<> controlPoint = coefficient.transpose();
226 gsWriteParaviewPoints(controlPoint, "CP_" + internal::to_string(k));
227
228 gsMatrix<> restrPar(2, indicesLS.size());
229 gsMatrix<> restrPoint(3, indicesLS.size());
230 for(size_t i = 0; i < indicesLS.size(); i++)
231 {
232 restrPar.col(i) = parameters.col(indicesLS[i]);
233 restrPoint.col(i) = points.col(indicesLS[i]);
234 }
235 gsWriteParaviewPoints(restrPar, "parameters_fun" + internal::to_string(k));
236 gsWriteParaviewPoints(restrPoint, "points_fun" + internal::to_string(k));
237
238 int level = basis.levelOf(k);
239 gsTensorBSplineBasis<2> tensor_basis = *basis.getBases()[level];
240 gsMatrix<> coef_pol(tensor_basis.size(), 3);
241 coef_pol.setZero();
242 std::vector<index_t> vector_employed_functions = set_to_vector(employed_functions);
243 for( unsigned i = 0 ; i < vector_employed_functions.size(); i++)
244 coef_pol.row(vector_employed_functions[i]) = coefficients.row(i);
245
246 gsTensorBSpline<2> tens_bspline(tensor_basis, coef_pol);
247 gsMatrix<real_t> domain = basis.support(k);
248 gsWriteParaview<real_t>(tens_bspline, domain, "iteration" + std::to_string(iteration) + "restr_to" + std::to_string(cp_num), 10000);
249
250 gsTensorBSplineBasis<2> coarse_basis = *basis.getBases()[0];
251 gsMatrix<> coef_pol_ls(coarse_basis.size(), 3);
252 coef_pol_ls.setZero();
253 std::vector<index_t> vector_employed_functions_ls = set_to_vector(employed_functions_ls);
254 for( unsigned i = 0 ; i < vector_employed_functions_ls.size(); i++)
255 coef_pol_ls.row(vector_employed_functions_ls[i]) = coefficients_ls.row(i);
256
257 gsTensorBSpline<2> tens_bspline_ls(coarse_basis, coef_pol_ls);
258 gsWriteParaview(tens_bspline_ls, "iteration" + std::to_string(iteration) + "_LSsurfaceOf" + std::to_string(cp_num), 10000, false, true);
259 }
260 return coefficient;
261 }

271gsMatrix<> get_coefficients_print_LS_surface(const gsTHBSplineBasis<2>& basis,
272 const gsMatrix<>& parameters,
273 const gsMatrix<>& points,
274 const real_t sigma,
275 int iteration,
276 int cp_magnitude,
277 const index_t cp_num,
278 const int it_max,
279 bool circle = true
280 )
281{
282
283 gsMatrix<> coefficients(basis.size(), points.rows());
284
285 for (index_t k = 0; k != basis.size(); k++)
286 {
287 // LS - least squares
288 int degreeLS = 0;
289 std::vector<index_t> indicesLS;
290 int enlarging_nr = 0;
291
292 data_least_square_approx(basis, parameters, k, degreeLS, indicesLS, cp_magnitude, enlarging_nr,
293 circle);
294
295 gsMatrix<> coefficient = get_coefficient_pol(basis, parameters, points, k,
296 degreeLS, indicesLS, sigma, iteration, cp_num, it_max);
297
298 coefficients.row(k) = coefficient.row(0);
299 }
300
301 return coefficients;
302}
