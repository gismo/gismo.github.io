134 void bringBoxToLvl(std::vector<index_t>& box,unsigned lvl)
135 {
136 gsTHBSplineBasis<d, T>* basis = static_cast< gsTHBSplineBasis<d,T>* > (this->m_basis);
137 if(static_cast<index_t>(lvl)<box[0])
138 {
139 const tensorBasis & maxBasis = *(basis->getBases()[lvl]);
140
141 T val_low,val_high;
142 for(unsigned i = 0;i<m_constrainedBoxes.size();++i)
143 {
144 const unsigned tLvl = box[0];
145 const tensorBasis & tBasis = *(basis->getBases()[tLvl]);
146
147 if(tLvl>lvl)
148 continue;
149 for (unsigned dim = 0; dim != d; dim++)
150 {
151 const gsKnotVector<T> & tKv = tBasis.component(dim).knots();
152 val_low=*(tKv.ubegin()+(box[1+dim]));
153 val_high=*(tKv.ubegin()+(box[1+d+dim]));
154 const gsKnotVector<T> & maxKv = maxBasis.component(dim).knots();
155 box[1+dim] = maxKv.uFind(val_low).uIndex();
156 box[1+d+dim] = maxKv.uFind(val_high).uIndex();
157 if(val_high==*(tKv.uend()-1))
158 box[1+d+dim]++;
159 }
160 box[0]=lvl;
161 }
162 }
163 else if(static_cast<index_t>(lvl)>box[0])
164 {
165 unsigned lvlDiff = lvl-box[0];
166 for(unsigned i = 0;i<2*d;++i)
167 {
168 box[i+1]*=math::ipow(2,lvlDiff);

191void gsHFittingLvlConstrained<d, T>::appendBox(std::vector<index_t>& boxes,
192 std::vector<index_t>& cells,
193 const gsVector<T>& parameter)
194{
195 gsTHBSplineBasis<d, T>* basis = static_cast< gsTHBSplineBasis<d,T>* > (this->m_basis);
196 const index_t maxLvl = basis->maxLevel();
197 const tensorBasis & tBasis = *(basis->getBases()[maxLvl]);
198
199 // get a cell
200 gsVector<index_t, d> a_cell;
201
202 for (short_t dim = 0; dim != d; dim++)
203 {
204 const gsKnotVector<T> & kv = tBasis.component(dim).knots();
205 a_cell(dim) = kv.uFind(parameter(dim)).uIndex();
206 }
207
208 // get level of a cell
209 gsVector<index_t, d> a_cell_upp = a_cell + gsVector<index_t, d>::Ones();
210 const index_t cell_lvl = basis->tree().query3(a_cell, a_cell_upp, maxLvl) + 1;
211
212 if (isCellAlreadyInserted(a_cell, cells))
213 return;
214
215 append(cells, a_cell);
216
217 // get the box
218 std::vector<index_t> box(2 * d + 1);
219 box[0] = cell_lvl;
220 for (short_t dim = 0; dim != d; dim++)
221 {
222 const index_t numBreaks = basis->numBreaks(cell_lvl, dim) - 1 ;
223
224 index_t lowIndex = 0;
225 if (cell_lvl < maxLvl)
226 {
227 const index_t shift = maxLvl - cell_lvl;
228 lowIndex = (a_cell(dim) >> shift);
229 //=(static_cast<util::make_unsigned<index_t>::type>(a_cell(dim)) >> shift);
230 }
231 else
232 {
233 const index_t shift = cell_lvl - maxLvl;
234 lowIndex = (a_cell(dim) << shift);
235 }
236
237 // apply extensions
238 index_t ext = m_ext[dim];
239 index_t low = ( (lowIndex > ext) ? (lowIndex - ext) : 0 );
240 index_t upp = ( (lowIndex + ext + 1 < numBreaks) ? (lowIndex + ext + 1) : numBreaks );
241
242 box[1 + dim ] = low;
243 box[1 + d + dim] = upp;
244 }
245
246 if (isBoxViolatingConstrainedBoxes(box))
247 return;
248
249 for(size_t i = 0;i<box.size();++i)
250 boxes.push_back(box[i]);
251}
