146  // look at eval_into
147  void fastEval_into(const gsMatrix<T>& u,
148  gsMatrix<T>& result) const
149  {
150  gsMatrix<index_t> indices;
151  this->active_into(u, indices);
152 
153  result.setZero(indices.rows(), u.cols());
154 
155  const unsigned maxLvl = this->m_tree.getMaxInsLevel() + 1;
156  std::vector< gsMatrix<T> > tmpResults(maxLvl, gsMatrix<T>());
157  std::vector< gsMatrix<index_t> > tmpActive(maxLvl, gsMatrix<index_t>());
158  gsVector<int> processed(maxLvl);
159 
160  for (int pt = 0; pt != u.cols(); pt++)
161  {
162  processed.setZero();
163 
164  for (int ind = 0; ind != indices.rows(); ind++)
165  {
166  unsigned index = indices(ind, pt);
167  if (ind != 0 && index == 0)
168  break;
169 
170  unsigned lvl = getPresLevelOfBasisFun(index);
171 
172  if (processed(lvl) == 0)
173  {
174  this->m_bases[lvl]->eval_into(u.col(pt), tmpResults[lvl]);
175  this->m_bases[lvl]->active_into(u.col(pt), tmpActive[lvl]);
176  processed(lvl) = 1;
177  }
178 
179  if (m_is_truncated[index] == -1)
180  {
181  index_t flatTenIndx = this->flatTensorIndexOf(index, lvl);
182  int localIndex = -1;
183  for (int row = 0; row != tmpActive[lvl].rows(); ++row)
184  {
185  if (tmpActive[lvl](row, 0) == flatTenIndx)
186  {
187  localIndex = row;
188  break;
189  }
190  }
191  result(ind, pt) = tmpResults[lvl](localIndex, 0);
192  }
193  else // basis function is truncated
194  {
195  const gsMatrix<T>& basis = tmpResults[lvl];
196  const gsMatrix<index_t>& active = tmpActive[lvl];
197 
198 
199  const gsSparseVector<T>& coefs = getCoefs(index);
200  T tmp = coefs(active(0, 0)) * basis(0, 0);
201  for (int i = 1; i < active.rows(); i++)
202  {
203  tmp += coefs(active(i, 0)) * basis(i, 0);
204  }
205  result(ind, pt) = tmp;
206  }
207  }
208  }
209  }
