52gsMatrix<> get_coefficients_periodic(const gsTHBSplineBasis<2>& thbBasis,

60 const int circle,
61 std::vector<index_t>& enlargement)
62{
63
64 //coefficients of the THB fit.
65 gsMatrix<> coefficients(thbBasis.size(), points.rows());
66 for (index_t k = 0; k < thbBasis.size(); k++)
67 {
68 gsMatrix<> coefficient = get_coefficient_not_scaled_stiff(thbBasis, parameters, points, iteration, k, cp_magnitude, hole, lambda, circle, enlargement);
69 coefficients.row(k) = coefficient.row(0);
70 }
71
72
73 // composite basis
74 std::vector< gsHTensorBasis<2>* > vectorOfBasis;
75 gsTHBSplineBasis<2> copy(thbBasis);
76 vectorOfBasis.push_back(&copy);
77 gsMPBESHSplineBasis<2, real_t> periodicBasis(vectorOfBasis, get_periodic_topology(dir));
78
79 gsMatrix<> global_coefs;
80 periodicBasis.local_coef_to_global_coef(coefficients, global_coefs);
81
82 gsMatrix<> result;
83 periodicBasis.global_coef_to_local_coef(global_coefs, result);
