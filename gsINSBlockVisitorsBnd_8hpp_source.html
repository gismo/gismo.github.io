71template <class T>
72void gsINSBlockVisitorRobinPCD<T>::assemble(gsDomainIterator<T>& element,
73 const gsMapData<T> & mapData,
74 const gsVector<T>& quWeights)
75{
76 for (index_t k = 0; k < quWeights.rows(); ++k) // loop over quadrature nodes
77 {
78 // Compute the outer normal vector on the side
79 outerNormal(mapData, k, m_side, m_unormal);
80
81 // Multiply quadrature weight by the geometry measure
82 const T weight = quWeights[k] * m_unormal.norm();
83
84 // Compute the unit normal vector
85 m_unormal.normalize();
86
87 T coef = - m_solUVals.col(k).transpose() * m_unormal;
88
89 m_localMat.noalias() += weight * coef * (m_bVals.col(k) * m_bVals.col(k).transpose());
90 }
91}

159template <class T>
160void gsINSBlockAVisitorNitsche<T>::assemble(gsDomainIterator<T>& element,
161 const gsMapData<T> & mapData,
162 const gsVector<T>& quWeights)
163{
164 gsMatrix<T> & bGrads = m_basisData[1];
165
166
167 for (index_t k = 0; k < quWeights.rows(); ++k) // loop over quadrature nodes
168 {
169
170 const gsMatrix<T> bVals = m_basisData[0].col(k);
171
172 // Compute the outer normal vector on the side
173 outerNormal(mapData, k, m_side, m_unormal);
174
175 // Multiply quadrature weight by the geometry measure
176 const T weight = quWeights[k] * m_unormal.norm();
177
178 // Compute the unit normal vector
179 m_unormal.normalize();
180
181 // Compute physical gradients at k as a Dim x NumActive matrix
182 transformGradients(mapData, k, bGrads, m_physGrad);
183
184 // Get penalty parameter
185 const T h = element.getCellSize();
186 const T gamma = m_penalty / (0 != h ? h : 1);
187
188 // Sum up quadrature point evaluations
189 m_localRhs.noalias() -= weight * m_viscosity * ((m_physGrad.transpose() * m_unormal - gamma * bVals)
190 * m_dirData.col(k).transpose());
191
192 m_localMat.noalias() -= weight * m_viscosity * (bVals * m_unormal.transpose() * m_physGrad
193 + (bVals * m_unormal.transpose() * m_physGrad).transpose()
194 - gamma * bVals * bVals.transpose());
195 }
