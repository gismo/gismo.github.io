80  break;

83  getKnots<2,T,gsTensorNurbsBasis>(basis,tensorKnots);
84  break;
85  case 3:
86  getKnots<3,T,gsTensorBSplineBasis>(basis,tensorKnots);
87  getKnots<3,T,gsTensorNurbsBasis>(basis,tensorKnots);
88  break;
89  default:
90  GISMO_ERROR("Basis type not understood");
91  break;
92  }
93 
94  std::vector<size_t> sizes(DIM);
95  std::vector<size_t> strides(DIM);
96  for (size_t d=0; d!=DIM; d++)
97  sizes[d] = tensorKnots[d].size();
98 
99  strides[0]=0;
100  for (size_t d=1; d!=DIM; d++)
101  strides[d]= strides[d-1]+sizes[d-1];
102 
103  gsMatrix<T> knots(DIM,std::accumulate(sizes.begin(),sizes.end(),0));
104  knots.setConstant(std::nan("1"));
105  for (size_t d=0; d!=DIM; d++)
106  knots.block(d,strides[d],1,sizes[d]) = tensorKnots[d].asMatrix();
107 
108  return knots;

184 template <class T>
185 std::tuple<gsMatrix<T>, gsMatrix<T>, std::vector<index_t>, std::vector<index_t>> packMultiPatch(const gsMultiPatch<T> &mp)
186 {
187  std::vector<gsMatrix<T>> knotMatrices;
188  knotMatrices.reserve(mp.nPatches());
189  std::vector<gsMatrix<T>> coefMatrices;
190  coefMatrices.reserve(mp.nPatches());
191  std::vector<index_t> knotCols;
192  std::vector<index_t> coefCols;
193 
194  for (typename gsMultiPatch<T>::const_iterator patch = mp.begin(); patch != mp.end(); ++patch) {
195  // Dereference the patch pointer to access its members
196  knotMatrices.push_back(knotsToMatrix((*patch)->basis()));
197  coefMatrices.push_back((*patch)->coefs().transpose());
198  knotCols.push_back(knotMatrices.back().cols());
199  coefCols.push_back(coefMatrices.back().cols());
200  }
201 
202  index_t knotRows = mp.domainDim();
203  index_t coefRows = mp.targetDim();
204  index_t knotColsSum = 0;
205  index_t coefColsSum = 0;
206 
207  for (size_t p = 0; p != mp.nPatches(); ++p) {
208  knotColsSum += knotMatrices[p].cols();
209  coefColsSum += coefMatrices[p].cols();
210  }
211 
212  gsMatrix<T> knotMatrix(knotRows, knotColsSum);
213  gsMatrix<T> coefMatrix(coefRows, coefColsSum);
214 
215  size_t currentKnotCol = 0;
216  size_t currentCoefCol = 0;
217 
218  for (size_t p = 0; p != mp.nPatches(); ++p) {
219  const auto& km = knotMatrices[p];
220  const auto& cm = coefMatrices[p];
221 
222  // Validate block dimensions before assignment
223  if (currentKnotCol + km.cols() <= knotColsSum && km.rows() == knotRows &&
224  currentCoefCol + cm.cols() <= coefColsSum && cm.rows() == coefRows) {
225  knotMatrix.block(0, currentKnotCol, knotRows, km.cols()) = km;
226  coefMatrix.block(0, currentCoefCol, coefRows, cm.cols()) = cm;
227 
228  currentKnotCol += km.cols();
229  currentCoefCol += cm.cols();
230  } else {
231  std::cerr << "Invalid block parameters for patch " << p << std::endl;
232  throw std::logic_error("Block parameters out of range");
233  }
234  }
235 
236  return std::make_tuple(knotMatrix, coefMatrix, knotCols, coefCols);
237 }

253 gsMultiPatch<T> unpackMultiPatch(const gsMatrix<T> &knotMatrix, const gsMatrix<T> &coefMatrix, const std::vector<index_t> &knotCols, const std::vector<index_t> &coefCols) {
254  gsMultiPatch<T> mp;

259  gsMatrix<T> km = knotMatrix.block(0, currentKnotCol, knotMatrix.rows(), knotCols[p]);
260  gsMatrix<T> cm = coefMatrix.block(0, currentCoefCol, coefMatrix.rows(), coefCols[p]).transpose();

267 
268  // Add the gsGeometry object to gsMultiPatch
269  mp.addPatch(std::move(geom));
270 
271  currentKnotCol += knotCols[p];
272  currentCoefCol += coefCols[p];
273  }
274 
275  return mp;
