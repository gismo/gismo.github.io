36  const index_t nc = M.cols();
37  for (index_t r=0; r!=nr; ++r)
38  {
39  piv = 0;

41  if (piv == nc ) continue;
42

46  M.col(piv).tail(br) * M.row(r).tail(bc) / M(r, piv);

48  }
